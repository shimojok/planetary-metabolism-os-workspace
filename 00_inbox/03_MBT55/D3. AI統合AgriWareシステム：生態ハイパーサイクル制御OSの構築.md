class MetabolicFlowAgent:
    def monitor_cycles(self, element_data):
        """元素循環の閉鎖性評価"""
        cycles = {}
        
        for element in ['C', 'N', 'P', 'K', 'Fe', 'Mn']:
            # 循環率の計算（入力vs出力）
            cycle_rate = self.calc_cycling_rate(
                element_data.inputs[element],
                element_data.outputs[element],
                element_data.internal_cycles[element]
            )
            
            # 漏出ポイントの特定
            leakage_points = self.identify_leakage_points(
                element_data.flow_pathways[element]
            )
            
            # 時間的安定性（季節変動対応）
            temporal_stability = self.assess_temporal_stability(
                element_data.time_series[element]
            )
            
            cycles[element] = {
                'cycling_rate': cycle_rate,
                'leakage_points': leakage_points,
                'temporal_stability': temporal_stability,
                'optimization_potential': 1.0 - cycle_rate  # 改善余地
            }
        
        return cycles
