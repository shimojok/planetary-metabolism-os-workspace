### **src/core/m3_simulator/__init__.py**
```python
"""
M³-BioSynergy Core Simulator
Implementation of the microbial-metabolic-modular synergy model
"""

import numpy as np
from typing import Dict, List, Tuple
from dataclasses import dataclass
from datetime import datetime

@dataclass
class MicrobialSpecies:
    """Individual microbial species data"""
    id: int
    name: str
    metabolic_type: str  # 'aerobic', 'anaerobic', 'facultative'
    growth_rate: float
    decay_rate: float
    functional_group: str  # 'decomposer', 'nitrogen_fixer', 'methanotroph', etc.

class M3BioSynergySimulator:
    """
    Core simulator implementing the M³-BioSynergy theory
    
    Features:
    1. 120-species microbial dynamics
    2. Metabolic network modeling
    3. Carbon/Nitrogen cycling
    4. GHG reduction calculation
    """
    
    def __init__(self, species_count: int = 120):
        self.species_count = species_count
        self.species = self._initialize_species()
        self.interaction_matrix = self._initialize_interactions()
        self.metabolic_matrix = self._initialize_metabolic_network()
        
        # State tracking
        self.current_population = np.ones(species_count) * 1000  # Initial population
        self.carbon_sequestered = 0.0
        self.ghg_reduction = 0.0
        
    def _initialize_species(self) -> List[MicrobialSpecies]:
        """Initialize 120 microbial species with realistic parameters"""
        species = []
        
        # Functional group distribution (based on MBT55 composition)
        groups = {
            'decomposer': 40,          # 40 species
            'nitrogen_fixer': 20,      # 20 species
            'methanotroph': 15,        # 15 species
            'denitrifier': 15,         # 15 species
            'phosphate_solubilizer': 10, # 10 species
            'heavy_metal_reducer': 10, # 10 species
            'hormone_producer': 10     # 10 species
        }
        
        species_id = 0
        for group, count in groups.items():
            for i in range(count):
                species.append(
                    MicrobialSpecies(
                        id=species_id,
                        name=f"{group}_{i}",
                        metabolic_type=self._assign_metabolic_type(group),
                        growth_rate=self._random_growth_rate(group),
                        decay_rate=self._random_decay_rate(group),
                        functional_group=group
                    )
                )
                species_id += 1
                
        return species
    
    def _assign_metabolic_type(self, group: str) -> str:
        """Assign metabolic type based on functional group"""
        if group in ['decomposer', 'methanotroph']:
            return 'aerobic'
        elif group in ['denitrifier', 'heavy_metal_reducer']:
            return 'anaerobic'
        else:
            return 'facultative'
    
    def _random_growth_rate(self, group: str) -> float:
        """Generate realistic growth rates"""
        base_rates = {
            'decomposer': 0.15,
            'nitrogen_fixer': 0.10,
            'methanotroph': 0.12,
            'denitrifier': 0.08,
            'phosphate_solubilizer': 0.11,
            'heavy_metal_reducer': 0.09,
            'hormone_producer': 0.13
        }
        return np.random.normal(base_rates[group], 0.02)
    
    def _initialize_interactions(self) -> np.ndarray:
        """Create species interaction matrix (Lotka-Volterra extended)"""
        matrix = np.zeros((self.species_count, self.species_count))
        
        # Symbiotic relationships (+ values)
        symbiotic_pairs = [
            ('decomposer', 'nitrogen_fixer'),
            ('nitrogen_fixer', 'hormone_producer'),
            ('phosphate_solubilizer', 'decomposer')
        ]
        
        # Competitive relationships (- values)
        competitive_pairs = [
            ('decomposer', 'decomposer'),  # Within group competition
            ('methanotroph', 'denitrifier')  # Oxygen competition
        ]
        
        # Apply interactions
        for i, sp_i in enumerate(self.species):
            for j, sp_j in enumerate(self.species):
                if i == j:
                    matrix[i, j] = -0.1  # Self-limitation
                else:
                    # Check for specific interactions
                    interaction = self._determine_interaction(sp_i, sp_j)
                    matrix[i, j] = interaction
        
        return matrix
    
    def _determine_interaction(self, sp1: MicrobialSpecies, sp2: MicrobialSpecies) -> float:
        """Determine interaction coefficient between two species"""
        # Symbiotic cases
        if (sp1.functional_group == 'decomposer' and sp2.functional_group == 'nitrogen_fixer'):
            return 0.05  # Decomposers provide carbon to nitrogen fixers
        if (sp1.functional_group == 'nitrogen_fixer' and sp2.functional_group == 'hormone_producer'):
            return 0.03  # Nitrogen fixers provide nitrogen to hormone producers
        
        # Competitive cases
        if sp1.functional_group == sp2.functional_group:
            return -0.02  # Intra-group competition
        
        # Neutral by default
        return 0.0
    
    def _initialize_metabolic_network(self) -> Dict:
        """Initialize metabolic network for GHG calculations"""
        return {
            'carbon_pathways': {
                'cellulose_decomposition': {'species': [0, 1, 2], 'efficiency': 0.85},
                'lignin_decomposition': {'species': [3, 4, 5], 'efficiency': 0.70},
                'humification': {'species': [6, 7, 8], 'efficiency': 0.45}
            },
            'nitrogen_pathways': {
                'nitrogen_fixation': {'species': [20, 21, 22], 'efficiency': 0.65},
                'denitrification': {'species': [45, 46, 47], 'efficiency': 0.82}
            },
            'ghg_pathways': {
                'methane_oxidation': {'species': [60, 61, 62], 'rate': 0.80},
                'nitrous_oxide_reduction': {'species': [63, 64, 65], 'rate': 0.72}
            }
        }
    
    def simulate_step(self, 
                     current_population: np.ndarray = None,
                     mbt_intervention: np.ndarray = None,
                     environmental_conditions: Dict = None,
                     timestep_hours: float = 1.0) -> Dict:
        """
        Simulate one time step of ecosystem dynamics
        
        Args:
            current_population: Current population of each species
            mbt_intervention: MBT55 intervention vector (enhancement factors)
            environmental_conditions: Temperature, pH, moisture, etc.
            timestep_hours: Duration of time step in hours
            
        Returns:
            Dictionary with simulation results
        """
        # Use current state if not provided
        if current_population is None:
            current_population = self.current_population.copy()
        
        if mbt_intervention is None:
            mbt_intervention = np.ones(self.species_count)
        
        if environmental_conditions is None:
            environmental_conditions = {
                'temperature': 25.0,
                'ph': 6.5,
                'moisture': 0.6,
                'organic_matter': 3.0
            }
        
        # Apply environmental factors
        env_factor = self._calculate_environmental_factor(environmental_conditions)
        
        # Calculate population dynamics (Lotka-Volterra extended)
        interaction_effect = self.interaction_matrix @ current_population
        intrinsic_growth = np.array([s.growth_rate for s in self.species])
        
        # MBT55 intervention enhances growth and interactions
        growth_rates = intrinsic_growth * mbt_intervention * env_factor
        
        # Population change equation
        delta_population = (growth_rates * current_population * 
                          (1 + 0.1 * interaction_effect) * 
                          timestep_hours / 24.0)
        
        # Apply decay
        decay_rates = np.array([s.decay_rate for s in self.species])
        delta_population -= decay_rates * current_population * timestep_hours / 24.0
        
        # Update population
        new_population = current_population + delta_population
        new_population = np.maximum(new_population, 0)  # No negative populations
        
        # Calculate metabolic outputs
        carbon_sequestration = self._calculate_carbon_sequestration(new_population)
        ghg_reduction = self._calculate_ghg_reduction(new_population)
        
        # Update state
        self.current_population = new_population
        self.carbon_sequestered += carbon_sequestration
        self.ghg_reduction += ghg_reduction
        
        return {
            'timestamp': datetime.now(),
            'population': new_population,
            'population_change': delta_population,
            'carbon_sequestration_kg': carbon_sequestration,
            'total_carbon_sequestered_kg': self.carbon_sequestered,
            'ghg_reduction_kg_co2e': ghg_reduction,
            'total_ghg_reduction_kg_co2e': self.ghg_reduction,
            'ecosystem_health': self._calculate_ecosystem_health(new_population),
            'hypercycle_stability': self._assess_hypercycle_stability(new_population)
        }
    
    def _calculate_environmental_factor(self, conditions: Dict) -> float:
        """Calculate environmental impact on microbial activity"""
        # Temperature effect (optimal at 25-35°C)
        temp = conditions['temperature']
        temp_factor = np.exp(-0.5 * ((temp - 30) / 10) ** 2)
        
        # pH effect (optimal at 6.5-7.5)
        ph = conditions['ph']
        ph_factor = np.exp(-0.5 * ((ph - 7.0) / 1.0) ** 2)
        
        # Moisture effect (optimal at 60-80%)
        moisture = conditions['moisture']
        moisture_factor = np.exp(-0.5 * ((moisture - 0.7) / 0.2) ** 2)
        
        return temp_factor * ph_factor * moisture_factor
    
    def _calculate_carbon_sequestration(self, population: np.ndarray) -> float:
        """Calculate carbon sequestration based on microbial activity"""
        # Carbon Use Efficiency (CUE) model
        total_biomass = np.sum(population)
        
        # Different groups have different CUE
        cue_factors = {
            'decomposer': 0.30,
            'nitrogen_fixer': 0.45,
            'methanotroph': 0.25,
            'denitrifier': 0.20,
            'phosphate_solubilizer': 0.35,
            'heavy_metal_reducer': 0.28,
            'hormone_producer': 0.40
        }
        
        total_cue = 0.0
        for i, species in enumerate(self.species):
            total_cue += population[i] * cue_factors[species.functional_group]
        
        # Convert to kg of carbon sequestered
        # Assuming 1e6 cells produce 1 mg of stable carbon per day
        carbon_kg = total_cue * 1e-6 * 24  # Convert to daily rate
        
        return carbon_kg
    
    def _calculate_ghg_reduction(self, population: np.ndarray) -> float:
        """Calculate GHG reduction (CO2e) from microbial activity"""
        ghg_reduction = 0.0
        
        # Methane oxidation by methanotrophs
        methanotroph_indices = [i for i, s in enumerate(self.species) 
                              if s.functional_group == 'methanotroph']
        methanotroph_population = np.sum(population[methanotroph_indices])
        
        # Each methanotroph cell oxidizes ~10^-6 g CH4/hour
        ch4_oxidized = methanotroph_population * 1e-6 * 25  # 25x GWP of CO2
        ghg_reduction += ch4_oxidized
        
        # N2O reduction by denitrifiers
        denitrifier_indices = [i for i, s in enumerate(self.species) 
                             if s.functional_group == 'denitrifier']
        denitrifier_population = np.sum(population[denitrifier_indices])
        
        # Each denitrifier reduces ~10^-7 g N2O/hour
        n2o_reduced = denitrifier_population * 1e-7 * 298  # 298x GWP of CO2
        ghg_reduction += n2o_reduced
        
        # Convert to kg CO2e
        return ghg_reduction * 1e-3  # Convert grams to kg
    
    def _calculate_ecosystem_health(self, population: np.ndarray) -> float:
        """Calculate overall ecosystem health score (0-100)"""
        # 1. Diversity score
        diversity = self._calculate_diversity(population)
        
        # 2. Functional balance score
        balance = self._calculate_functional_balance(population)
        
        # 3. Activity level
        activity = np.sum(population) / (self.species_count * 10000)
        activity_score = min(100, activity * 20)
        
        # Combine scores
        health_score = (diversity * 0.4 + balance * 0.4 + activity_score * 0.2)
        
        return min(100, max(0, health_score))
    
    def _calculate_diversity(self, population: np.ndarray) -> float:
        """Calculate Shannon diversity index"""
        total = np.sum(population)
        if total == 0:
            return 0.0
        
        proportions = population[population > 0] / total
        shannon = -np.sum(proportions * np.log(proportions))
        
        # Normalize to 0-100 (max diversity for 120 species ~ ln(120) ≈ 4.79)
        return (shannon / 4.79) * 100
    
    def _calculate_functional_balance(self, population: np.ndarray) -> float:
        """Calculate balance between functional groups"""
        group_populations = {}
        for group in set(s.functional_group for s in self.species):
            indices = [i for i, s in enumerate(self.species) 
                      if s.functional_group == group]
            group_populations[group] = np.sum(population[indices])
        
        total = sum(group_populations.values())
        if total == 0:
            return 0.0
        
        # Calculate coefficient of variation (lower is more balanced)
        proportions = [p/total for p in group_populations.values()]
        mean_prop = np.mean(proportions)
        std_prop = np.std(proportions)
        
        if mean_prop == 0:
            return 0.0
        
        cv = std_prop / mean_prop
        
        # Convert to score (0-100, higher is better)
        balance_score = 100 * np.exp(-cv)
        
        return balance_score
    
    def _assess_hypercycle_stability(self, population: np.ndarray) -> Dict:
        """Assess the stability of the ecological hypercycle"""
        stability_indicators = {}
        
        # 1. Self-amplification check
        total_growth = np.sum([s.growth_rate * population[i] 
                              for i, s in enumerate(self.species)])
        total_decay = np.sum([s.decay_rate * population[i] 
                             for i, s in enumerate(self.species)])
        
        stability_indicators['amplification_ratio'] = total_growth / total_decay if total_decay > 0 else 1.0
        
        # 2. Functional redundancy
        functional_redundancy = self._calculate_functional_redundancy(population)
        stability_indicators['functional_redundancy'] = functional_redundancy
        
        # 3. Interaction network connectivity
        connectivity = self._calculate_network_connectivity(population)
        stability_indicators['network_connectivity'] = connectivity
        
        # Overall stability score
        stability_score = (
            min(1.5, stability_indicators['amplification_ratio']) * 40 +
            stability_indicators['functional_redundancy'] * 30 +
            stability_indicators['network_connectivity'] * 30
        ) / 100
        
        stability_indicators['overall_stability'] = min(100, stability_score * 100)
        
        # Classification
        if stability_indicators['overall_stability'] >= 80:
            stability_indicators['stability_class'] = 'STABLE'
        elif stability_indicators['overall_stability'] >= 60:
            stability_indicators['stability_class'] = 'MODERATE'
        else:
            stability_indicators['stability_class'] = 'UNSTABLE'
        
        return stability_indicators
    
    def _calculate_functional_redundancy(self, population: np.ndarray) -> float:
        """Calculate functional redundancy (0-1)"""
        active_groups = set()
        for i, species in enumerate(self.species):
            if population[i] > 0:
                active_groups.add(species.functional_group)
        
        total_groups = len(set(s.functional_group for s in self.species))
        return len(active_groups) / total_groups
    
    def _calculate_network_connectivity(self, population: np.ndarray) -> float:
        """Calculate interaction network connectivity (0-1)"""
        # Count active interactions
        active_species = [i for i, p in enumerate(population) if p > 0]
        
        if len(active_species) < 2:
            return 0.0
        
        total_possible_interactions = len(active_species) * (len(active_species) - 1)
        active_interactions = 0
        
        for i in active_species:
            for j in active_species:
                if i != j and abs(self.interaction_matrix[i, j]) > 0.001:
                    active_interactions += 1
        
        return active_interactions / total_possible_interactions if total_possible_interactions > 0 else 0.0

# 基本的なシミュレーションデモ
# このノートブックはGitHub上で直接表示されます

import sys
sys.path.append('../src')

from core.m3_simulator import M3BioSynergySimulator
import matplotlib.pyplot as plt
import numpy as np

# シミュレーターの作成
simulator = M3BioSynergySimulator(species_count=120)

# 24時間シミュレーション
results = []
for hour in range(24):
    result = simulator.simulate_step(timestep_hours=1.0)
    results.append(result)

# 結果の可視化
fig, axes = plt.subplots(2, 2, figsize=(12, 8))

# 炭素隔離の推移
axes[0, 0].plot([r['carbon_sequestration_kg'] for r in results])
axes[0, 0].set_title('Carbon Sequestration per Hour')
axes[0, 0].set_xlabel('Hour')
axes[0, 0].set_ylabel('kg CO2')

# GHG削減の推移
axes[0, 1].plot([r['ghg_reduction_kg_co2e'] for r in results])
axes[0, 1].set_title('GHG Reduction per Hour')
axes[0, 1].set_xlabel('Hour')
axes[0, 1].set_ylabel('kg CO2e')

# 生態系健康スコア
axes[1, 0].plot([r['ecosystem_health'] for r in results])
axes[1, 0].set_title('Ecosystem Health Score')
axes[1, 0].set_xlabel('Hour')
axes[1, 0].set_ylabel('Score (0-100)')

# ハイパーサイクル安定性
axes[1, 1].plot([r['hypercycle_stability']['overall_stability'] for r in results])
axes[1, 1].set_title('Hypercycle Stability')
axes[1, 1].set_xlabel('Hour')
axes[1, 1].set_ylabel('Stability Score')

plt.tight_layout()
plt.show()

# サマリー表示
final = results[-1]
print(f"24時間の総合計:")
print(f"総炭素隔離量: {final['total_carbon_sequestered_kg']:.2f} kg CO2")
print(f"総GHG削減量: {final['total_ghg_reduction_kg_co2e']:.2f} kg CO2e")
print(f"最終生態系健康スコア: {final['ecosystem_health']:.1f}/100")
print(f"ハイパーサイクル安定性: {final['hypercycle_stability']['stability_class']}")
```

name: CI/CD Pipeline

on:
  push:

    - name: Upload coverage
