import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from typing import Dict, List, Tuple

class AfricaMBTSimulation:
    def __init__(self):
        self.setup_africa_parameters()
        
    def setup_africa_parameters(self):
        """アフリカ全土のパラメーターを設定"""
        
        # アフリカ基本統計データ
        self.africa_params = {
            # 人口統計
            'total_population': 1.4e9,           # 14億人
            'urban_population_ratio': 0.45,      # 都市化率45%
            'rural_population_ratio': 0.55,      # 農村人口率55%
            
            # 家畜データ (FAO統計ベース)
            'cattle_population': 350e6,          # 3.5億頭
            'sheep_population': 400e6,           # 4億頭
            'goat_population': 400e6,            # 4億頭
            'poultry_population': 2e9,           # 20億羽
            
            # 農業データ
            'agricultural_land': 1e9,            # 10億ha
            'degraded_land': 300e6,              # 3億ha（劣化農地）
            'cropland_area': 250e6,              # 2.5億ha（耕作地）
            
            # 経済パラメーター
            'gdp_per_capita': 2000,              # 1人当たりGDP [USD]
            'waste_treatment_cost': 3000,        # 廃棄物処理コスト [円/トン]
            'compost_price': 5000,               # 堆肥価格 [円/トン]
            'carbon_price': 1500,                # 炭素価格 [円/tCO₂e]
            
            # スケーリング係数（ケニアからの拡張）
            'scale_from_kenya': 25,              # 人口比 1400/56 = 25
        }
        
        # MBT性能パラメーター（ナイロビシミュレーションから継承）
        self.mbt_params = {
            'compost_production_rate': 0.4,
            'methane_emission_factor': 0.06,
            'gwp_100': 25,
            'carbon_sequestration_rate': 0.15,
            'co2_conversion_factor': 3.67,
            'fertilizer_replacement_rate': 0.05,
            'fertilizer_production_co2': 5,
            'livestock_methane_reduction': 0.3,
            'livestock_methane_emission': 0.07,
            'food_loss_avoidance_factor': 2.0,
            'biomass_increase_rate': 0.5,
        }
        
        # アフリカ廃棄物発生原単位
        self.waste_generation_rates = {
            'urban_food_waste': 0.4,    # kg/人/日（都市部）
            'rural_food_waste': 0.2,    # kg/人/日（農村部）
            'livestock_waste_cattle': 15,  # kg/頭/日（牛）
            'livestock_waste_sheep_goat': 1,  # kg/頭/日（羊・ヤギ）
            'livestock_waste_poultry': 0.1, # kg/羽/日（家禽）
            'agri_residue_rate': 0.25,  # 農業生産量の25%が残渣
            'sewage_sludge_urban': 0.1, # kg/人/日（都市部下水汚泥）
        }
    
    def calculate_africa_waste_generation(self) -> Dict:
        """アフリカ全土の廃棄物発生量を計算"""
        
        p = self.africa_params
        w = self.waste_generation_rates
        
        waste_volumes = {}
        
        # 1. 食品廃棄物・食品ロス
        urban_food_waste = p['total_population'] * p['urban_population_ratio'] * w['urban_food_waste'] * 365 / 1000
        rural_food_waste = p['total_population'] * p['rural_population_ratio'] * w['rural_food_waste'] * 365 / 1000
        waste_volumes['food_waste'] = urban_food_waste + rural_food_waste
        
        # 2. 家畜排泄物
        cattle_waste = p['cattle_population'] * w['livestock_waste_cattle'] * 365 / 1000
        sheep_goat_waste = (p['sheep_population'] + p['goat_population']) * w['livestock_waste_sheep_goat'] * 365 / 1000
        poultry_waste = p['poultry_population'] * w['livestock_waste_poultry'] * 365 / 1000
        waste_volumes['livestock_waste'] = cattle_waste + sheep_goat_waste + poultry_waste
        
        # 3. 農産物残渣（農業生産量の25%と仮定）
        # アフリカの穀物生産量約2億トンから推計
        base_agricultural_production = 800e6  # 8億トン（総農業生産）
        waste_volumes['agricultural_residue'] = base_agricultural_production * w['agri_residue_rate']
        
        # 4. 下水汚泥
        waste_volumes['sewage_sludge'] = p['total_population'] * p['urban_population_ratio'] * w['sewage_sludge_urban'] * 365 / 1000
        
        # 5. 加工食品廃棄物（食品廃棄物の20%と仮定）
        waste_volumes['processing_waste'] = waste_volumes['food_waste'] * 0.2
        
        # 総廃棄物量
        waste_volumes['total'] = sum([waste_volumes[k] for k in waste_volumes if k != 'total'])
        
        return waste_volumes
    
    def calculate_ghg_reduction_sources(self, waste_volumes: Dict) -> Dict:
        """10項目のGHG削減ソースを計算"""
        
        p = self.africa_params
        m = self.mbt_params
        
        compost_produced = waste_volumes['total'] * m['compost_production_rate']
        
        ghg_reduction = {}
        
        # 1. 廃棄物処理回避によるメタン発生削減
        methane_avoided = waste_volumes['total'] * m['methane_emission_factor']
        ghg_reduction['waste_methane_avoidance'] = methane_avoided * m['gwp_100']
        
        # 2. 堆肥による炭素固定
        carbon_sequestered = compost_produced * m['carbon_sequestration_rate']
        ghg_reduction['carbon_sequestration'] = carbon_sequestered * m['co2_conversion_factor']
        
        # 3. 化学肥料削減による製造時のCO2削減
        fertilizer_replaced = compost_produced * m['fertilizer_replacement_rate']
        ghg_reduction['fertilizer_production_reduction'] = fertilizer_replaced * m['fertilizer_production_co2']
        
        # 4. 家畜のメタン排出削減（MBT55微生物群による）
        # 反芻動物（牛、羊、ヤギ）のみ対象
        ruminant_population = p['cattle_population'] + p['sheep_population'] + p['goat_population']
        livestock_methane_reduced = ruminant_population * m['livestock_methane_emission'] * m['livestock_methane_reduction']
        ghg_reduction['livestock_methane_reduction'] = livestock_methane_reduced * m['gwp_100']
        
        # 5. 食品ロス削減による排出削減
        food_waste_reduced = waste_volumes['food_waste'] * 0.5  # 50%削減想定
        ghg_reduction['food_loss_reduction'] = food_waste_reduced * m['food_loss_avoidance_factor']
        
        # 6. 農地修復によるバイオマス増加
        land_rehabilitated = min(compost_produced / 10, p['degraded_land'])  # 堆肥10トン/ha/年
        ghg_reduction['biomass_carbon_accumulation'] = land_rehabilitated * m['biomass_increase_rate'] * m['co2_conversion_factor']
        
        # 7. 輸送距離短縮による排出削減
        # 地域分散処理による輸送距離50%削減、輸送排出原単位0.2 kgCO₂/トンkm、平均距離50km想定
        transport_reduction = waste_volumes['total'] * 0.5 * 0.2 * 50 / 1000
        ghg_reduction['transportation_reduction'] = transport_reduction
        
        # 8. 化石燃料代替による排出削減（バイオガス利用）
        # 廃棄物の anaerobic digestion によるエネルギー回収
        biogas_potential = waste_volumes['total'] * 0.1  # 100 m³/トン、0.6 kgCO₂e/m³替代
        ghg_reduction['fossil_fuel_replacement'] = biogas_potential * 0.6
        
        # 9. 土壌の窒素循環改善によるN2O削減
        # 化学肥料使用削減によるN2O排出削減（N2O GWP = 298）
        nitrogen_fertilizer_replaced = fertilizer_replaced * 0.3  # 肥料の30%が窒素
        n2o_reduction = nitrogen_fertilizer_replaced * 0.01 * 298  # 1%がN2Oとして放出
        ghg_reduction['n2o_reduction'] = n2o_reduction
        
        # 10. 都市ヒートアイランド効果軽減によるエネルギー削減
        # コンポスト適用による緑化面積増加で冷房需要削減
        urban_area_improved = land_rehabilitated * 0.1  # 修復農地の10%が都市近郊
        ghg_reduction['urban_heat_island_reduction'] = urban_area_improved * 0.5  # 0.5 tCO₂e/ha/年
        
        # 総GHG削減量
        ghg_reduction['total_reduction'] = sum(ghg_reduction.values())
        
        return ghg_reduction
    
    def calculate_economic_analysis(self, waste_volumes: Dict, ghg_reduction: Dict) -> Dict:
        """経済性分析（投資対効果、グリーンプレミアム）"""
        
        p = self.africa_params
        compost_produced = waste_volumes['total'] * self.mbt_params['compost_production_rate']
        
        economic = {}
        
        # 収益計算
        economic['treatment_cost_savings'] = waste_volumes['total'] * p['waste_treatment_cost']
        economic['compost_sales_revenue'] = compost_produced * p['compost_price']
        economic['carbon_credit_revenue'] = ghg_reduction['total_reduction'] * p['carbon_price']
        
        # その他便益
        economic['fertilizer_savings'] = compost_produced * 0.05 * 80000  # 化学肥料代替
        economic['water_savings'] = compost_produced * 10 * 0.3 * 50  # 節水効果
        
        # 総収益
        economic['total_revenue'] = sum(economic.values())
        
        # 投資コスト
        investment_per_ton = 50000  # 円/トン処理能力
        economic['initial_investment'] = waste_volumes['total'] * investment_per_ton
        
        # 運用コスト（投資額の15%）
        economic['annual_om_costs'] = economic['initial_investment'] * 0.15
        
        # 正味現在価値計算（20年間、割引率5%）
        project_lifetime = 20
        discount_rate = 0.05
        
        annual_net_cashflow = economic['total_revenue'] - economic['annual_om_costs']
        npv = sum([annual_net_cashflow / (1 + discount_rate)**t for t in range(1, project_lifetime + 1)])
        economic['npv'] = npv - economic['initial_investment']
        
        # 投資対効果
        economic['roi_percentage'] = (economic['npv'] / economic['initial_investment']) * 100
        
        # グリーンプレミアム計算
        conventional_system_cost = waste_volumes['total'] * 5000  # 従来システムコスト
        green_premium_cost = economic['annual_om_costs'] + (economic['initial_investment'] / project_lifetime)
        economic['green_premium_ratio'] = (economic['total_revenue'] - green_premium_cost) / green_premium_cost
        
        # 回収期間
        economic['payback_period'] = economic['initial_investment'] / annual_net_cashflow
        
        return economic

def run_africa_scale_simulation():
    """アフリカ全土シミュレーション実行"""
    
    simulator = AfricaMBTSimulation()
    
    # 廃棄物量計算
    waste_volumes = simulator.calculate_africa_waste_generation()
    
    # GHG削減量計算
    ghg_reduction = simulator.calculate_ghg_reduction_sources(waste_volumes)
    
    # 経済性分析
    economic_analysis = simulator.calculate_economic_analysis(waste_volumes, ghg_reduction)
    
    return {
        'waste_volumes': waste_volumes,
        'ghg_reduction': ghg_reduction,
        'economic_analysis': economic_analysis,
        'simulator': simulator
    }

def create_africa_visualizations(results):
