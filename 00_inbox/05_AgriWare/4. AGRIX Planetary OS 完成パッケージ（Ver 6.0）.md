import numpy as np

class AgrixPlanetaryOS:
    def __init__(self):
        self.mbt_library_v2 = "Active"
        self.nitrogenase_targets = {"Mo": 1.0, "Fe": 100.0, "Eh_min": -100, "Eh_max": 200}

    def run_diagnosis(self, sensor_data):
        # ニトロゲナーゼ活性化指数の算出 (NAI)
        nai = (sensor_data['mo'] / self.nitrogenase_targets['Mo']) * \
              (1 - abs(sensor_data['eh'] - 50) / 150)
        return max(0, min(1, nai))

    def calculate_pbpe(self, carbon_t, health_impact_score):
        # PBPEトークン生成ロジック
        base_value = carbon_t * 10 
        health_premium = health_impact_score * 1.5
        return base_value + health_premium

    def generate_mpp(self, nai_score):
        # 微生物処方パケット(MPP)の生成
        if nai_score < 0.5:
            return {"action": "Apply_HMT_Chelate", "mbt55_dose": "High"}
        return {"action": "Maintain_Cycle", "mbt55_dose": "Low"}
