# Ⅰ. 自動タグ抽出ロジック

既存JSONには：

- 主要生薬
    
- MBT代謝産物
    
- ターゲット疾患
    

があるため、文字列ルールで抽出可能。

```python
def auto_tag_extraction(formula):

    tags = []

    # 生薬由来
    herbs = " ".join(formula.get("主要生薬", []))
    metabolites = " ".join(formula.get("MBT代謝産物", []))

    if "配糖体" in metabolites or "ジンセノ" in metabolites:
        tags.append("配糖体")

    if "ポリフェノール" in metabolites:
        tags.append("ポリフェノール")

    if "精油" in metabolites:
        tags.append("精油")

    if "多糖" in metabolites:
        tags.append("多糖")

    if "アルカロイド" in metabolites or "エフェドリン" in metabolites:
        tags.append("アルカロイド")

    if "硫黄" in metabolites:
        tags.append("硫黄")

    return tags
```

これは骨格版なので単純ですが、後で高度化可能です。

# Ⅱ. 一括ベクトル化エンジン

```python
import json
import numpy as np
import pandas as pd
from sklearn.cluster import KMeans

# -----------------------------
# ルール辞書（前回確定版）
# -----------------------------
TAG_RULE = {...}
DISEASE_RULE = {...}
METABOLITE_RULE = {...}

vectorizer = MBT15Vectorizer(TAG_RULE, DISEASE_RULE, METABOLITE_RULE)

# -----------------------------
# パイプライン本体
# -----------------------------
def build_matrix(json_path):

    with open(json_path, 'r', encoding='utf-8') as f:
        data = json.load(f)

    formulas = data["Data_Payload"]

    matrix = []
    names = []

    for formula in formulas:

        # 自動タグ追加
        formula["Functional_Tags"] = auto_tag_extraction(formula)

        vec = vectorizer.vectorize(formula)

        matrix.append(vec)
        names.append(formula["Name"])

    matrix = np.array(matrix)

    return matrix, names
```

# Ⅳ. 初期クラスタリング（骨格）

```python
def run_clustering(matrix, k=8):

    kmeans = KMeans(n_clusters=k, random_state=42)
    labels = kmeans.fit_predict(matrix)

    return labels
```
