### setup_repository.py

```python
import os

def setup_healthbook_repository():
    # 1. Directory Structure Definition
    project_name = "HealthBook_MI_Core"
    folders = [
        project_name,
        f"{project_name}/core",
        f"{project_name}/data",
        f"{project_name}/docs",
        f"{project_name}/examples",
        f"{project_name}/cloud_integration"
    ]

    print(f"--- Initializing {project_name} Repository Structure ---")
    for folder in folders:
        if not os.path.exists(folder):
            os.makedirs(folder)
            print(f"[Created] {folder}")

    # 2. File Content Generation
    files_to_create = {
        # --- README.md (Vision & Strategy) ---
        f"{project_name}/README.md": """# HealthBook: Metabolic Intelligence (MI) Engine
**A Core Component of the Planetary Metabolism OS**

## Overview
HealthBook is an advanced metabolic inference engine that bridges human clinical data (300,000 cases) with the broader 'Planetary Immune System' vision. It translates lifestyle-induced metabolic imbalances into preventive phytochemical prescriptions.

## Key Features
- **Hamada-Method Engine:** 200-question high-resolution lifestyle analysis.
- **Shimojo-Phytochemical Matrix:** Defense-layer-based nutritional intervention.
- **Azure Ready:** Built for global scalability on Microsoft Azure.

## Developer Quick Start
Run `examples/demo_cli.py` to start the diagnostic interface.
""",

        # --- Core Engine (Logic) ---
        f"{project_name}/core/engine.py": """
import datetime

class MetabolicIntelligenceEngine:
    \"\"\"The MI Brain: Analyzing Metabolic Imbalance via Clinical Patterns.\"\"\"
    def __init__(self, database):
        self.db = database
        self.version = "1.0.0-Alpha"

    def analyze(self, responses):
        # Basic logic for mapping responses to risks
        score = sum(1 for r in responses.values() if r is True)
        return {
            "timestamp": datetime.datetime.now().isoformat(),
            "metabolic_load_index": score / len(responses) if responses else 0,
            "status": "Diagnostic Complete"
        }
""",

        # --- Data Placeholder ---
        f"{project_name}/data/questions_en.json": '{"1": "Do you often skip breakfast?", "2": "Do you prefer salty foods?"}',

        # --- Demo Script for Pydroid/CLI ---
        f"{project_name}/examples/demo_cli.py": """
import sys
import os

# Adjust path for local import
sys.path.append(os.path.abspath(os.path.join(os.path.dirname(__file__), '..')))
from core.engine import MetabolicIntelligenceEngine

def run():
    print("HealthBook MI Demo v1.0")
    print("-" * 30)
    engine = MetabolicIntelligenceEngine(database=None)
    
    # Simple Mock Questions
    q1 = input("Do you skip breakfast? (y/n): ") == 'y'
    q2 = input("Do you like salty food? (y/n): ") == 'y'
    
    report = engine.analyze({1: q1, 2: q2})
    print(f"\\nAnalysis Result: {report}")

if __name__ == "__main__":
    run()
"""
    }

    # 3. Write files
    for path, content in files_to_create.items():
        with open(path, "w", encoding="utf-8") as f:
            f.write(content.strip())
        print(f"[Generated] {path}")

    print("-" * 50)
    print(f"SUCCESS: '{project_name}' is ready for GitHub.")
    print(f"Navigate to '{project_name}/examples' and run 'demo_cli.py' to test.")

if __name__ == "__main__":
    setup_healthbook_repository()

```
