### 1. Azure Ready Demo: `examples/azure_ready_demo.py`

このスクリプトは、生成した `knowledge_base_full_en.json` を読み込み、クラウド連携を模した高度な分析フローを実行します。

```python
"""
examples/azure_ready_demo.py
Pre-configured for Microsoft Azure Integration.
Demonstrates the Planetary Metabolism OS as a Scalable Cloud Infrastructure.
"""

import json
import os
import time
import uuid
from core.engine import MetabolicIntelligenceEngine

class AzureMetabolicCloudInterface:
    def __init__(self):
        self.device_id = f"HB-NODE-{uuid.uuid4().hex[:8].upper()}"
        self.engine = MetabolicIntelligenceEngine(database=None) # Will load the JSON

    def simulate_telemetry_upload(self, data):
        """Simulates secure data transmission to Azure IoT Hub."""
        print(f"\\n[Azure IoT] Connecting to: metabolic-os.azure-devices.net...")
        time.sleep(1)
        print(f"[Azure IoT] Sending telemetry for Device: {self.device_id}")
        print(f"[Azure IoT] Payload encrypted and transmitted.")

    def run_cloud_analysis(self, user_responses):
        # Local MI processing
        print("\\n[MI Engine] Running high-resolution metabolic analysis...")
        report = self.engine.analyze(user_responses)
        
        # Simulating cloud-side value conversion (PBPE: Performance-Based Personal Economy)
        report["health_token_generated"] = round(report["metabolic_load_index"] * 10, 2)
        
        self.simulate_telemetry_upload(report)
        return report

def main():
    print("="*65)
    print("HealthBook MI: Azure Cloud Integration Demo")
    print("="*65)

    # Mock Input (Simulating a user with high risk markers)
    # Question 1: Skip Breakfast, Question 5: Processed Food, Question 51: No Veggies
    user_inputs = {1: True, 5: True, 51: True, 101: True}
    
    interface = AzureMetabolicCloudInterface()
    final_report = interface.run_cloud_analysis(user_inputs)

    print("\\n" + "-"*20 + " CLOUD ANALYTICS RESULT " + "-"*20)
    print(f"Device ID      : {interface.device_id}")
    print(f"Metabolic Load : {final_report['metabolic_load_index']:.2f} (Warning: High)")
    print(f"Wealth Impact  : {final_report['health_token_generated']} METAB-Tokens generated via Preventive Action")
    print("-" * 65)
    print("Strategic Note: This node is ready to sync with Planetary Metabolism OS.")

if __name__ == "__main__":
    main()
```
