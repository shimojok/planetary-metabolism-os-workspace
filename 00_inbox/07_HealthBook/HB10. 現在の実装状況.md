def display_detailed_analysis(results: Dict):
    """詳細分析結果の表示"""
    print("\n" + "="*80)
    print("詳細分析結果")
    print("="*80)
    
    risk_analysis = results.get("risk_analysis", {})
    
    if "basic_risk" in risk_analysis:
        print("\n📈 基本リスク評価:")
        for disease, info in risk_analysis["basic_risk"].items():
            score = info.get("score", 0)
            matched = info.get("matched_factors", 0)
            total = info.get("total_factors", 1)
            print(f"   • {disease}: {score:.1%} ({matched}/{total}因子)")
    
    if "adjusted_risk" in risk_analysis:
        print("\n🔄 調整後リスク:")
        for disease, info in risk_analysis["adjusted_risk"].items():
            adj_score = info.get("adjusted_score", 0)
            orig_score = info.get("score", 0)
            if adj_score != orig_score:
                change = ((adj_score - orig_score) / orig_score * 100) if orig_score > 0 else 0
                print(f"   • {disease}: {orig_score:.1%} → {adj_score:.1%} ({change:+.1f}%)")
    
    print(f"\n🎯 信頼度スコア: {risk_analysis.get('confidence_score', 0):.1%}")
