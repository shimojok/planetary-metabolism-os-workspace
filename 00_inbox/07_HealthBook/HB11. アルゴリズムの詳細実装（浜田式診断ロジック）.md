    def _evaluate_metabolic_profiles(self, responses: Dict[int, bool]) -> Dict[str, float]:
        """代謝プロファイルの評価"""
        profile_scores = {}
        
        for profile_name, profile_info in self.metabolic_profiles.items():
            total_weight = 0
            matched_questions = 0
            
            for q_id in profile_info["key_questions"]:
                if q_id in responses and responses[q_id]:
                    # 質問の有意性を考慮
                    significance = self.question_significance.get(q_id, {"weight": 0.5})
                    total_weight += significance["weight"]
                    matched_questions += 1
            
            # スコア計算（マッチした質問の割合 × 平均重み）
            if profile_info["key_questions"]:
                question_ratio = matched_questions / len(profile_info["key_questions"])
                avg_weight = total_weight / matched_questions if matched_questions > 0 else 0
                score = question_ratio * avg_weight
                
                # 重症度分類
                thresholds = profile_info["severity_thresholds"]
                if score >= thresholds["重度"]:
                    severity = "重度"
                elif score >= thresholds["中度"]:
                    severity = "中度"
                elif score >= thresholds["軽度"]:
                    severity = "軽度"
                else:
                    severity = "正常範囲"
                
                profile_scores[profile_name] = {
                    "score": round(score, 3),
                    "severity": severity,
                    "matched_questions": matched_questions,
                    "total_questions": len(profile_info["key_questions"]),
                    "key_findings": self._extract_key_findings(profile_name, responses)
                }
        
        return profile_scores
    
    def _evaluate_disease_risks(self, responses: Dict[int, bool], 
                               metabolic_scores: Dict) -> Dict[str, Dict]:
        """疾病リスクの詳細評価"""
        disease_risks = {}
        
        for disease, pattern in self.disease_patterns.items():
            # 基本スコア計算
            base_score = self._calculate_disease_base_score(disease, responses)
            
            # 代謝プロファイルによる調整
            metabolic_profile = pattern["metabolic_profile"]
            if metabolic_profile in metabolic_scores:
                metabolic_impact = metabolic_scores[metabolic_profile]["score"]
                adjusted_score = base_score * (1 + metabolic_impact * 0.5)
            else:
                adjusted_score = base_score
            
            # サブタイプの特定
            subtype = self._identify_subtype(disease, responses)
            
            # リスクレベル分類
            risk_level = self._classify_risk_level(adjusted_score)
            
            # 介入ポイントの優先度付け
            intervention_priority = self._prioritize_interventions(disease, responses, metabolic_scores)
            
            disease_risks[disease] = {
                "base_score": round(base_score, 3),
                "adjusted_score": round(min(adjusted_score, 1.0), 3),
                "risk_level": risk_level,
                "subtype": subtype,
                "metabolic_basis": pattern["metabolic_profile"],
                "key_indicators": self._extract_key_indicators(disease, responses),
                "intervention_priority": intervention_priority,
                "predicted_response": pattern.get("response_prediction", {})
            }
        
        return disease_risks
    
    def _calculate_disease_base_score(self, disease: str, responses: Dict[int, bool]) -> float:
        """疾病の基本スコア計算"""
        pattern = self.disease_patterns[disease]
        question_weights = pattern["key_indicators"]["question_weights"]
        
        total_weight = 0
        matched_weight = 0
        
        for q_id, weight in question_weights.items():
            total_weight += weight
            if q_id in responses and responses[q_id]:
                matched_weight += weight
        
        return matched_weight / total_weight if total_weight > 0 else 0
    
    def _identify_subtype(self, disease: str, responses: Dict[int, bool]) -> str:
        """疾病サブタイプの特定"""
        pattern = self.disease_patterns[disease]
        
        # 質問パターンに基づくサブタイプ分類

            # ナトリウム感受性 vs ストレス性 vs 遺伝性

            stress_score = sum(1 for q in stress_questions if q in responses and responses[q])
            
            if salt_score >= 2:
                return "ナトリウム感受性型"
            elif stress_score >= 2:
                return "ストレス性型"
            else:
                return "その他の型"
        
        elif disease == "糖尿病":
            # インスリン抵抗性 vs インスリン分泌不全
            insulin_resistance_questions = [6, 8, 45]  # 肥満・運動不足関連

                                 if q in responses and responses[q])
            secretion_score = sum(1 for q in secretion_questions 
                                if q in responses and responses[q])

                return "インスリン抵抗性型"
            elif secretion_score >= 2:
                return "インスリン分泌不全型"
            else:
                return "混合型"
        
        return pattern.get("subtype", "標準型")
    
    def _classify_risk_level(self, score: float) -> str:
        """リスクレベルの分類"""
        if score >= 0.8:

            return "高いリスク"

            return "中等度リスク"
        elif score >= 0.2:
            return "軽度リスク"

        pattern = self.disease_patterns[disease]

        # 質問回答に基づく優先度調整
        for intervention in interventions:

            if target == "ナトリウム摂取":

                salt_count = sum(1 for q in salt_questions if q in responses and responses[q])

    def _find_common_questions(self, profiles: List[Tuple], responses: Dict[int, bool]) -> List[int]:
        """共通の質問項目を検出"""
        common_questions = []
        
        for profile_name, score in profiles:
            profile_info = self.metabolic_profiles[profile_name]
            for q_id in profile_info["key_questions"]:
                if q_id in responses and responses[q_id]:
                    common_questions.append(q_id)
        
        # 出現頻度順にソート
        from collections import Counter
        freq = Counter(common_questions)
        return [q_id for q_id, count in freq.most_common(5)]

    def _generate_intervention_strategy(self, imbalance_patterns: List[Dict],
                                       disease_risks: Dict) -> Dict:

        strategy = {
            "priority_levels": {},
            "timeline": {},
            "monitoring_points": []
        }
        
        # 高リスク疾病の介入優先度
        high_risk_diseases = [
            (disease, data) 
            for disease, data in disease_risks.items() 
            if data["risk_level"] in ["非常に高いリスク", "高いリスク"]
        ]
        
        for disease, data in high_risk_diseases[:3]:  # 上位3つ
            strategy["priority_levels"][disease] = {
                "level": "高優先度",
                "actions": data["intervention_priority"],
                "expected_timeline": "1-3ヶ月での改善目標",
                "success_indicators": data.get("predicted_response", {})
            }
        
        # 代謝不全パターンへの対応
        for pattern in imbalance_patterns:
            pattern_type = pattern["type"]
            strategy["priority_levels"][pattern_type] = {
                "level": "中優先度",
                "actions": pattern.get("recommendations", []),
                "expected_timeline": "3-6ヶ月での改善目標"
            }
        
        # モニタリングポイントの設定
        strategy["monitoring_points"] = [
            {"interval": "2週間後", "check": "食事記録の確認"},
            {"interval": "1ヶ月後", "check": "症状の変化評価"},
            {"interval": "3ヶ月後", "check": "生活習慣改善の定着度"}
        ]
        
        return strategy
    
    def _adjust_with_lab_data(self, disease_risks: Dict, lab_data: Dict[str, float]) -> Dict:
        """検査データによるリスク調整"""
        adjusted = disease_risks.copy()
        
        for disease, data in adjusted.items():
            adjustment_factor = 1.0
            
            if disease == "高血圧" and "血圧" in lab_data:
                bp = lab_data["血圧"]
                if bp > 140:
                    adjustment_factor *= 1.3
                elif bp > 130:
                    adjustment_factor *= 1.1
            
            elif disease == "糖尿病" and "血糖" in lab_data:
                glucose = lab_data["血糖"]
                if glucose > 126:
                    adjustment_factor *= 1.4
                elif glucose > 110:
                    adjustment_factor *= 1.2
            
            data["lab_adjusted_score"] = min(data["adjusted_score"] * adjustment_factor, 1.0)
            data["lab_adjustment_factor"] = adjustment_factor
        
        return adjusted
    
    def _calculate_confidence(self, responses: Dict[int, bool], 
                             lab_data: Dict = None) -> float:
        """分析の信頼度計算"""
        confidence = 0.0
        
        # 問診回答の完全性
        answered_count = len([v for v in responses.values() if v is not None])
        completeness = answered_count / 50  # サンプル50問で計算
        confidence += completeness * 0.6
        
        # 検査データの有無
        if lab_data and len(lab_data) >= 3:
            confidence += 0.3
        
        # 回答の一貫性チェック
        consistency = self._check_response_consistency(responses)
        confidence += consistency * 0.1
        
        return min(confidence, 1.0)
    
    def _check_response_consistency(self, responses: Dict[int, bool]) -> float:
        """回答の一貫性チェック"""
        # 関連する質問間の一貫性を評価
        consistency_pairs = [
            ([1, 25], 0.8),  # 朝食抜きと間食の関連性
            ([6, 45], 0.7),  # 甘い物と夜食の関連性
            ([8, 89], 0.6)   # 運動不足とストレスの関連性
        ]
        
        matches = 0
        total_pairs = 0
        
        for question_ids, expected_correlation in consistency_pairs:
            values = [responses.get(q, None) for q in question_ids]
            if all(v is not None for v in values):
                total_pairs += 1
                # すべてTrueまたはすべてFalseなら一貫性あり
                if len(set(values)) == 1:
                    matches += 1
        
        return matches / total_pairs if total_pairs > 0 else 0.5
