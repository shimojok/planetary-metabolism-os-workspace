```python
"""
浜田式診断 超簡易デモ（実行テスト用）
"""

class SimpleHamadaDemo:
    """浜田式診断の超簡易版"""
    
    def simple_analysis(self, responses):
        """簡易分析"""
        print("\n浜田式簡易診断結果:")
        print("="*40)
        
        # 糖代謝チェック
        sugar_questions = [1, 6, 25, 67]  # 朝食抜き、甘い物、間食、清涼飲料
        sugar_score = sum(1 for q in sugar_questions if responses.get(q, False))
        
        if sugar_score >= 3:
            print("⚠️ 糖代謝異常の疑い:")
            print("  • 朝食を必ず摂る")
            print("  • 甘い物を控える")
            print("  • 間食を減らす")
        
        # 塩分代謝チェック
        salt_questions = [3, 25, 67]  # 塩辛い物、加工食品、外食
        salt_score = sum(1 for q in salt_questions if responses.get(q, False))
        
        if salt_score >= 2:
            print("\n⚠️ ナトリウム代謝負荷:")
            print("  • 塩分控えめ（1日6g未満）")
            print("  • 加工食品を減らす")
            print("  • カリウム豊富な食品を摂る")
        
        # 総合評価
        total_risk = sugar_score + salt_score
        
        if total_risk >= 4:
            print(f"\n🔴 高リスク（リスク点数: {total_risk}）")
            print("専門家の相談を推奨します")
        elif total_risk >= 2:
            print(f"\n🟡 中リスク（リスク点数: {total_risk}）")
            print("生活習慣改善を推奨します")
        else:
            print(f"\n🟢 低リスク（リスク点数: {total_risk}）")
            print("現在の習慣を維持しましょう")

# 実行テスト
def quick_test():
    """クイックテスト"""
    print("浜田式診断 クイックテスト")
    print("回答例: はい=1, いいえ=0")
    
    responses = {}
    questions = [
        (1, "朝食を抜くことが多い"),

        ans = input(f"{question}？ (1/0): ").strip()

    demo = SimpleHamadaDemo()
```
