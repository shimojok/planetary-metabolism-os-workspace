### ステップ2：統合推論エンジンの実装（Python）

次に、このJSONを読み込んで、個人のリスクと改善策を導き出す「脳」を構築します。

```python
import json

class BioNexusInferenceEngine:
    def __init__(self, knowledge_base_path):
        with open(knowledge_base_path, 'r', encoding='utf-8') as f:
            self.kb = json.load(f)
            
    def analyze_phenotype(self, user_responses, agrix_quality_data):
        """
        user_responses: 200項目の回答データ
        agrix_quality_data: 摂取している作物の「質」データ（下條分類に基づく）
        """
        disease_scores = {d: 0 for d in self.kb['Disease_Weights'].keys()}
        
        # 1. 浜田式リスク・マッピング
        for q_id, is_yes in user_responses.items():
            if is_yes:
                question = next(q for q in self.kb['Questions'] if q['id'] == q_id)
                for disease in question['risk_factors']:
                    disease_scores[disease] += 1.0 # 重み付けの基本加算
        
        # 2. 下條式ファイトケミカル・ブースト（リスク相殺）
        for disease, score in disease_scores.items():
            mitigation_type = self.kb['Disease_Weights'][disease]['Mitigation_Group']
            
            # AGRIXの高品質な「質」が、特定部位の防衛能力を高める
            if mitigation_type == "Hydrophilic_Defense":
                # 水溶性成分（フラボノイド等）が豊富なら、血液・体液系疾患リスクを低減
                score *= (1.0 - agrix_quality_data['hydrophilic_index'])
            elif mitigation_type == "Lipophilic_Defense":
                # 脂溶性成分（カロテノイド等）が豊富なら、細胞膜系疾患リスクを低減
                score *= (1.0 - agrix_quality_data['lipophilic_index'])
                
            disease_scores[disease] = score

        return self.generate_prescription_report(disease_scores)

    def generate_prescription_report(self, scores):
        # リスクが高い順にソートし、MBT55処方と漢方294を紐付けて出力
        return sorted(scores.items(), key=lambda x: x[1], reverse=True)

# 実行イメージ
engine = BioNexusInferenceEngine('knowledge_base.json')
# AGRIX産（高品質）のファイトケミカルを摂取している設定
```
