## より実用的な方法：既存JSONから変換

既存のマスターJSON（オブジェクト形式）をお持ちであれば、以下の変換スクリプトが最も簡単です：

```python
"""
既存JSONを配列形式に変換するスクリプト
"""

import json

def convert_to_array_format(input_file: str, output_file: str):
    """オブジェクト形式のJSONを配列形式に変換"""
    
    # 入力ファイル読み込み
    with open(input_file, 'r', encoding='utf-8') as f:
        data = json.load(f)
    
    # オブジェクト形式から配列形式に変換
    formulas_obj = data['mbt_kampo_library']['formulas']
    formulas_array = []
    
    for fid, fdata in formulas_obj.items():
        formula_entry = {"id": fid, **fdata}
        formulas_array.append(formula_entry)
    
    # IDでソート
    formulas_array.sort(key=lambda x: x['id'])
    
    # 出力データ作成（配列形式）
    output_data = {
        "mbt_kampo_library": {
            "version": data['mbt_kampo_library']['version'],
            "last_updated": data['mbt_kampo_library']['last_updated'],
            "total_formulas": data['mbt_kampo_library']['total_formulas'],
            "fully_defined": len(formulas_array),
            "classification_source": data['mbt_kampo_library']['classification_source'],
            "mbt55_version": data['mbt_kampo_library']['mbt55_version'],
            "formulas": formulas_array  # 配列！
        }
    }
    
    # 保存
    with open(output_file, 'w', encoding='utf-8') as f:
        json.dump(output_data, f, ensure_ascii=False, indent=2)
    
    print(f"✅ 変換完了: {output_file}")
    print(f"   方剤数: {len(formulas_array)}")

if __name__ == "__main__":
    # 使用例
    convert_to_array_format(
        input_file="mbt_kampo_library_master.json",  # 既存のオブジェクト形式JSON
        output_file="mbt_kampo_library_array.json"    # 出力先（配列形式）
    )
```

この変換スクリプトを使えば、お持ちの既存JSONから簡単に配列形式を生成できます。
