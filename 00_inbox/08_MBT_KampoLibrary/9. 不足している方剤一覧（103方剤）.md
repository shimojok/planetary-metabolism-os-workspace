        formulas[fid]["name"] = formulas[src_id]["name"] + f" (複製)"

# メタデータの更新

root["fully_defined"] = len([f for f in formulas.values() if "components" in f and f["components"]])
root["last_updated"] = datetime.now().strftime("%Y-%m-%d")

# 保存
with open(output_file, "w", encoding="utf-8") as f:
    json.dump(data, f, ensure_ascii=False, indent=2)

print(f"✅ 完了: {output_file}")
print(f"   → 元の処方数: 191")
print(f"   → 追加後の処方数: {len(formulas)}")
