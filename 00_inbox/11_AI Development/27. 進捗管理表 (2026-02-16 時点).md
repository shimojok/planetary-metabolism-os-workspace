Python

```
class MedicalCostImpactModel:
    """
    Simulates the reduction of healthcare costs through MI-based prevention.
    """
    def __init__(self, region_population):
        self.population = region_population
        self.avg_cost_per_person = 5000 # USD annual
        self.prevention_efficacy = 0.30 # 30% reduction target

    def calculate_annual_saving(self):
        total_current_cost = self.population * self.avg_cost_per_person
        potential_saving = total_current_cost * self.prevention_efficacy
        return {
            "Total_Regional_Saving_USD": potential_saving,
            "Per_Capita_Benefit_USD": self.avg_cost_per_person * self.prevention_efficacy
        }
```
