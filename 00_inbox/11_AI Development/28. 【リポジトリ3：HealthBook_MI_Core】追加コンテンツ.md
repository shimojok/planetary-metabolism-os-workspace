#### 5. demo/mbt_kanpo_demo.py (漢方・スクリーニング体験デモ)

このプログラムは、ユーザーの代謝データ（フェノタイプ）を入力すると、MBT漢方ライブラリーから最適な成分をスクリーニングし、さらにその成分が「どのMBT55農場で、どのような土壌状態で生産されたか」までをAGRIXと連動して追跡するシミュレーターです。

Python

```
import json
import time

class MBT_Kanpo_Demo_System:
    """
    Demo: MBT Food & Herbal Screening System
    Integrates Metabolic Intelligence with the Phytochemical Library.
    """
    def __init__(self):
        # Sample Database: MBT Enhanced Phytochemicals
        self.library = [
            {"id": "MBT-H01", "name": "Ginsenoside-Rb1 (Enhanced)", "target": "Fatigue/Metabolism", "origin_soil": "MBT-Farm-A"},
            {"id": "MBT-H02", "name": "Curcumin (High-Absorption)", "target": "Inflammation", "origin_soil": "MBT-Farm-B"},
            {"id": "MBT-H03", "name": "Berberine (Metabolic-Sync)", "target": "Glucose Control", "origin_soil": "MBT-Farm-C"}
        ]

    def run_phenotype_screening(self, user_health_data):
        print("\n[MI Engine] Analyzing Metabolic Phenotype...")
        time.sleep(1)
        
        # Simple screening logic based on glucose or inflammation
        recommendations = []
        if user_health_data.get("glucose_level", 0) > 110:
            recommendations.append(self.library[2]) # Berberine
        if user_health_data.get("inflammation_marker", 0) > 5:
            recommendations.append(self.library[1]) # Curcumin
            
        return recommendations

    def display_demo(self, user_name, health_data):
        print(f"--- Welcome to BioNexus 3 Demo: {user_name} ---")
        results = self.run_phenotype_screening(health_data)
        
        if not results:
            print("Status: Metabolism Balanced. Maintaining with standard MBT diet.")
        else:
            print(f"Found {len(results)} Optimal Metabolic Match(es):")
            for item in results:
                print(f" >> COMPONENT: {item['name']}")
                print(f" >> TARGET: {item['target']}")
                print(f" >> PRODUCED BY: {item['origin_soil']} (Verified via AGRIX PBPE)")
        print("\n--- Demo Execution Complete ---")

# Execute Demo
if __name__ == "__main__":

    user_data = {
        "glucose_level": 125,
        "inflammation_marker": 8
    }
    
    demo = MBT_Kanpo_Demo_System()
    demo.display_demo("Shimojo (Test User)", user_data)
```
