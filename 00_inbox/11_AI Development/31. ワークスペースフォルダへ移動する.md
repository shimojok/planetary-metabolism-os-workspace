### ステップ1：GitHubから直接URLリストを抽出する（Python）

以下のコードをコピーして、**`get_urls.py`** という名前で保存し、実行してください。

Python

```
import requests

# 設定
USER = "shimojok"
REPO = "planetary-metabolism-os-workspace"
BRANCH = "main"

def get_all_files(path=""):
    url = f"https://api.github.com/repos/{USER}/{REPO}/contents/{path}?ref={BRANCH}"
    response = requests.get(url)
    if response.status_code != 200:
        return []
    
    files = response.json()
    all_raw_urls = []
    
    for file in files:
        if file['type'] == 'dir':
            # フォルダの中身を再帰的に取得
            all_raw_urls.extend(get_all_files(file['path']))
        elif file['name'].endswith('.md'):
            # RAW URLを生成
            raw_url = f"https://raw.githubusercontent.com/{USER}/{REPO}/{BRANCH}/{file['path']}"
            all_raw_urls.append(raw_url)
    
    return all_raw_urls

# 実行
urls = get_all_files()
with open('workspace_all_urls_final.txt', 'w') as f:
    for url in urls:
        f.write(url + '\n')

print(f"完了！ {len(urls)} 件のURLを workspace_all_urls_final.txt に保存しました。")
```

### 1. `get_urls.py` を以下の内容に書き換えてください

（メモ帳で開き、中身をすべて消してから、以下を貼り付けて保存してください）

Python

```
import urllib.request
import json

# 設定
USER = "shimojok"
REPO = "planetary-metabolism-os-workspace"
BRANCH = "main"

def get_all_files(path=""):
    url = f"https://api.github.com/repos/{USER}/{REPO}/contents/{path}?ref={BRANCH}"
    print(f"Scanning: {path if path else 'root'}...")
    
    try:
        with urllib.request.urlopen(url) as response:
            files = json.loads(response.read().decode())
            all_raw_urls = []
            
            for file in files:
                if file['type'] == 'dir':
                    # フォルダの中身を再帰的に取得
                    all_raw_urls.extend(get_all_files(file['path']))
                elif file['name'].endswith('.md'):
                    # RAW URLを生成
                    raw_url = f"https://raw.githubusercontent.com/{USER}/{REPO}/{BRANCH}/{file['path']}"
                    all_raw_urls.append(raw_url)
            return all_raw_urls
    except Exception as e:
        print(f"Error accessing {path}: {e}")
        return []

# 実行
urls = get_all_files()
with open('workspace_all_urls_final.txt', 'w', encoding='utf-8') as f:
    for url in urls:
        f.write(url + '\n')

print(f"\n成功！ {len(urls)} 件のURLを workspace_all_urls_final.txt に保存しました。")
```
