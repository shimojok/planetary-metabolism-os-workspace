```javascript
// 環境インパクト計算
function calculateEnvironmentalImpact(organicWaste, mscUnits) {
  // 1. 廃棄物処理による直接GHG削減
  const wasteGHGReduction = organicWaste * 0.8;  // ton-CO2e/ton廃棄物
  
  // 2. MBT腐植質・発酵肥料生成量
  const humusProduction = organicWaste * 0.6;  // 変換効率60%
  const fertilizerProduction = organicWaste * 0.3;
  
  // 3. 土壌修復可能面積
  const soilRestorationArea = humusProduction / 5;  // 5 ton/ha必要
  
  // 4. 土壌炭素隔離（SOC増加）
  const carbonSequestration = soilRestorationArea * 2.2;  // ton-CO2e/ha/year
  
  // 5. 化学肥料代替によるGHG削減
  const fertilizerGHGReduction = fertilizerProduction * 3.5;  // 化学肥料製造のCO2原単位
  
  // 総GHG削減量
  const totalGHGReduction = 
    wasteGHGReduction + 
    carbonSequestration + 
    fertilizerGHGReduction;
  
  return {
    totalGHGReduction,      // ton-CO2e/year
    soilRestorationArea,    // hectares
    humusProduction,        // tons
    fertilizerProduction    // tons
  };
}
```

```javascript
function calculateROI(investment, farmerIncomeIncrease, carbonCredits) {
  // 1. 年間収益
  const annualRevenue = 
    farmerIncomeIncrease * 0.15 +  // プラットフォーム手数料15%
    carbonCredits * 25;             // 炭素クレジット価格$25/ton
  
  // 2. 運営コスト
  const operatingCost = investment * 0.20;  // 年20%
  
  // 3. 純利益
  const netProfit = annualRevenue - operatingCost;
  
  // 4. ROI
  const roi = (netProfit / investment) * 100;
  
  // 5. ペイバック期間
  const paybackPeriod = investment / netProfit;
  
  return { roi, paybackPeriod, netProfit, annualRevenue };
}
```

#### **D. **ネガティブ・グリーン・プレミアム**の計算**

```javascript
function calculateNegativeGreenPremium(mbtCost, conventionalCost, environmentalBenefit) {
  // 従来農業の総コスト（環境コスト含む）
  const conventionalTotal = 
    conventionalCost.production + 
    conventionalCost.environmental;  // 外部性
  
  // MBT再生農業の総コスト
  const mbtTotal = 
    mbtCost.production - 
    environmentalBenefit.carbonCredits;  // 炭素クレジット収益
  
  // グリーン・プレミアム（マイナスなら"ネガティブ"）
  const greenPremium = mbtTotal - conventionalTotal;
  const greenPremiumPercent = (greenPremium / conventionalTotal) * 100;
  
  return {
    greenPremium,          // USD（マイナスなら経済的優位）
    greenPremiumPercent,   // %
    message: greenPremium < 0 
      ? `ネガティブ・グリーン・プレミアム: ${Math.abs(greenPremiumPercent).toFixed(1)}% 安い！`
      : `グリーン・プレミアム: ${greenPremiumPercent.toFixed(1)}% 高い`
  };
}
```

```javascript
function calculateRequirements(targetGHG = 510000000) {
```

### **Phase 4: 組み合わせ最適化ゲーム**

```javascript
function ProbioticsScreeningGame() {
  const [selectedDisease, setSelectedDisease] = useState(null);
  const [selectedBacteria, setSelectedBacteria] = useState([]);
  const [results, setResults] = useState(null);
  
  // プレイヤーが菌株を選択
  const handleSelectBacteria = (bacterium) => {
    if (selectedBacteria.length < 10) {  // 最大10種選択可能
      setSelectedBacteria([...selectedBacteria, bacterium]);
    }
  };
  
  // 効果予測アルゴリズム
  const predictEfficacy = () => {
    const pathways = metabolicPathways[selectedDisease.id];
    let efficacyScore = 0;
    let coveredPathways = [];
    let synergy = 0;
    
    // 各代謝経路がカバーされているかチェック
    pathways.blockages.forEach(blockage => {
      const covering = selectedBacteria.filter(b => 
        blockage.requiredBacteria.some(req => b.species.includes(req))
      );
      
      if (covering.length > 0) {
        efficacyScore += blockage.severity === 'CRITICAL' ? 40 : 
                        blockage.severity === 'HIGH' ? 25 : 10;
        coveredPathways.push(blockage.pathway);
        
        // 複数菌がカバーする場合はシナジー
        if (covering.length > 1) {
          synergy += 15;
        }
      }
    });
    
    // 多様性ボーナス
    const diversityBonus = selectedBacteria.length * 2;
    
    // 総合スコア
    const totalScore = Math.min(100, efficacyScore + synergy + diversityBonus);
    
    return {
      totalScore,
      coveredPathways,
      synergy,
      clinicalPotential: totalScore > 70 ? 'HIGH' : totalScore > 40 ? 'MEDIUM' : 'LOW',
      recommendedNextSteps: totalScore > 70 
        ? ['臨床試験Phase 1開始', 'FDA申請準備']
        : ['追加菌株スクリーニング', 'in vitro試験']
    };
  };
  
  return (
    <div className="screening-game">
      {/* 疾患選択 */}
      <DiseaseSelector 
        diseases={diseases} 
        onSelect={setSelectedDisease} 
      />
      
      {selectedDisease && (
        <>
          {/* 代謝経路マップ */}
          <MetabolicPathwayMap 
            pathways={metabolicPathways[selectedDisease.id]} 
          />
          
          {/* 菌株ライブラリ */}
          <BacteriaLibrary 
            database={mbt55Database}
            selectedBacteria={selectedBacteria}
            onSelect={handleSelectBacteria}
          />
          
          {/* 予測ボタン */}
          <button onClick={() => setResults(predictEfficacy())}>
            効果を予測
          </button>
          
          {/* 結果表示 */}
          {results && (
            <ResultsPanel 
              results={results}
              selectedBacteria={selectedBacteria}
            />
          )}
        </>
      )}
    </div>
  );
}
```
