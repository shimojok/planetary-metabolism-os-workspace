# 2. models.py（データ構造）

```python
from dataclasses import dataclass
from typing import Dict


@dataclass
class SoilState:
    """土壌状態の基本パラメータ"""
    soc: float          # Soil Organic Carbon [t-C/ha]
    cn_ratio: float     # C/N 比
    ph: float
    moisture: float     # [%]
    temperature: float  # [°C]


@dataclass
class OrganicInput:
    """投入有機物の組成"""
    carbon: float       # [t-C/ha/year]
    nitrogen: float     # [t-N/ha/year]
    lignin_ratio: float
    lipid_ratio: float
    protein_ratio: float
    starch_ratio: float


@dataclass
class MBT55Params:
    """MBT55 のモデルパラメータ"""
    kd: float                 # 分解定数 k_d [1/year]
    lambda_cf: float          # 最大炭素固定率 λ
    km_ed: float              # 電子飽和定数 K_m
    eta: float                # 電子再固定効率 η (0–1)
    guild_activity: Dict[str, float]  # 各ギルドの活動強度（0–1）
```

# 3. electron_flow.py（電子フロー Jₑ）

```python
from .models import SoilState, OrganicInput, MBT55Params


def compute_electron_flow(
    soil: SoilState,
    org: OrganicInput,
    params: MBT55Params,
) -> float:
    """
    電子フロー J_e を計算する簡易モデル。
    ギルド別の基質量 × 活動強度 × 経路効率を足し合わせる。
    """

    substrate = {
        "starch": org.carbon * org.starch_ratio,
        "protein": org.carbon * org.protein_ratio,
        "lipid": org.carbon * org.lipid_ratio,
        "cellulose": org.carbon * org.lignin_ratio,
    }

    phi = {
        "starch": 1.0,
        "protein": 0.9,
        "lipid": 1.2,
        "cellulose": 0.7,
    }

    je = 0.0
    for guild, s in substrate.items():
        k_i = params.guild_activity.get(guild, 0.0)
        phi_i = phi.get(guild, 0.0)
        je += k_i * s * phi_i

    env_factor = _environment_factor(soil)
    return je * env_factor


def _environment_factor(soil: SoilState) -> float:
    """pH・水分・温度から環境係数を計算（簡易モデル）"""
    ph_factor = 1.0 - abs(7.0 - soil.ph) * 0.1
    moisture_factor = 0.5 + min(soil.moisture, 100.0) / 200.0
    temp_factor = 0.5 + max(0.0, min(soil.temperature, 40.0)) / 80.0
    return max(0.0, ph_factor * moisture_factor * temp_factor)
```

# 4. carbon_cycle.py（炭素固定効率 C_f と SOC 変化）

```python
from .models import SoilState, OrganicInput, MBT55Params
from .electron_flow import compute_electron_flow


def compute_carbon_fixation_efficiency(
    electron_donor_eff: float,
    params: MBT55Params,
) -> float:
    """
    C_f = λ * E_d / (E_d + K_m)
    """
    ed = max(electron_donor_eff, 0.0)
    return params.lambda_cf * ed / (ed + params.km_ed) if ed > 0 else 0.0


def compute_soc_delta(
    soil: SoilState,
    org: OrganicInput,
    params: MBT55Params,
) -> float:
    """
    dC_s/dt = I - k_d * C_s + f_m(MBT55) * η
    ここでは f_m を C_f とみなし、電子フローから計算する。
    """
    je = compute_electron_flow(soil, org, params)
    cf = compute_carbon_fixation_efficiency(je, params)

    dcs_dt = org.carbon - params.kd * soil.soc + cf * params.eta
    return dcs_dt
```

# 5. simulate.py（1年シミュレーション）

```python
from .models import SoilState, OrganicInput, MBT55Params
from .carbon_cycle import compute_soc_delta


def simulate_mbt55_cycle(
    soil_init: SoilState,
    org: OrganicInput,
    params: MBT55Params,
    years: int = 5,
    dt: float = 1.0,
):
    """
    MBT55 導入下での SOC の時系列を返す。
    """
    soc_list = []
    soil = soil_init

    for t in range(int(years / dt)):
        delta = compute_soc_delta(soil, org, params)
        soil = SoilState(
            soc=soil.soc + delta * dt,
            cn_ratio=soil.cn_ratio,
            ph=soil.ph,
            moisture=soil.moisture,
            temperature=soil.temperature,
        )
        soc_list.append(soil.soc)

    return soc_list
```
