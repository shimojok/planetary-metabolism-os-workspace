### HealthBook Core Logic (English Version) - Metabolic Intelligence Engine

```python
"""
HealthBook Platform: Metabolic Intelligence (MI) Core Engine v1.0
Integrating Hamada's Clinical Insights with Planetary Metabolism OS Vision.

This module functions as the 'Planetary Immune System' node, 
translating dietary/lifestyle patterns into metabolic risk and 
phytochemical defense prescriptions.
"""

import json
import datetime
from typing import Dict, List, Any

# =================================================================
# 1. METABOLIC KNOWLEDGE BASE (Simplified for Demo)
# =================================================================

class MetabolicKnowledgeBase:
    """
    Encapsulates the clinical wisdom of 300,000 diagnoses.
    Translates 'Hidden Meanings' into actionable biological data.
    """
    
    @staticmethod
    def get_diagnostic_questions() -> Dict[int, Dict[str, Any]]:
        # English translation of Hamada's 200-question logic (Sample subset)
        return {
            1: {
                "question": "Do you often skip breakfast?",
                "biological_impact": "Circadian Rhythm Disruption / Glucose Instability",
                "risk_tags": ["Metabolic Lag", "Insulin Resistance"],
                "target_diseases": ["Diabetes", "Obesity", "Chronic Fatigue"]
            },
            2: {
                "question": "Do you dislike dark green or brightly colored vegetables?",
                "biological_impact": "Phytochemical Deficiency / Oxidative Stress",
                "risk_tags": ["Antioxidant Gap"],
                "target_diseases": ["Cardiovascular Disease", "Cancer Risk", "Early Aging"]
            },
            3: {
                "question": "Do you prefer salty foods or add extra salt frequently?",
                "biological_impact": "Sodium-Potassium Imbalance / Cellular Dehydration",
                "risk_tags": ["Hypertension Risk", "Renal Strain"],
                "target_diseases": ["Hypertension", "Stomach Cancer", "Nephritis"]
            }
        }

    @staticmethod
    def get_phytochemical_matrix() -> Dict[str, Dict[str, Any]]:
        # Shimojo-Hamada Phytochemical Classification
        return {
            "Sulforaphane": {
                "source": "Broccoli Sprouts",
                "function": "Phase 2 Detoxification Enzyme Activation",
                "defense_layer": "Detoxification / Anti-inflammation"
            },
            "Lycopene": {
                "source": "Tomatoes",
                "function": "Singlet Oxygen Quenching (Strong Antioxidant)",
                "defense_layer": "Cardiovascular Protection"
            },
            "Anthocyanin": {
                "source": "Blueberries / Purple Cabbage",
                "function": "Vascular Repair / Retinal Support",
                "defense_layer": "Microvascular Integrity"
            }
        }

# =================================================================
# 2. INFERENCE ENGINE (The 'MI' Brain)
# =================================================================

class MetabolicIntelligenceEngine:
    """
    The core logic that identifies 'Metabolic Imbalance' from lifestyle patterns.
    """
    
    def __init__(self):
        self.db = MetabolicKnowledgeBase()
        self.user_responses = {}
        self.analysis_report = {}

    def analyze_imbalance(self, answers: Dict[int, bool]):
        """
        Translates Yes/No responses into Metabolic Risk Scores.
        """
        questions = self.db.get_diagnostic_questions()
        risk_scores = {}
        identified_threats = []
        
        for q_id, response in answers.items():
            if response is True:  # If the risk behavior exists
                q_data = questions.get(q_id)
                for disease in q_data["target_diseases"]:
                    risk_scores[disease] = risk_scores.get(disease, 0) + 0.25
                identified_threats.append(q_data["biological_impact"])
        
        self.analysis_report = {
            "timestamp": datetime.datetime.now().isoformat(),
            "metabolic_threats": list(set(identified_threats)),
            "disease_risk_profile": risk_scores,
            "status": "Analysis Complete"
        }
        return self.analysis_report

    def prescribe_phytochemicals(self) -> List[Dict]:
        """
        Matches disease risks with specific Phytochemical defense layers.
        """
        matrix = self.db.get_phytochemical_matrix()
        recommendations = []
        
        # Logic: If Cardiovascular risk is high, recommend Lycopene, etc.
        risks = self.analysis_report.get("disease_risk_profile", {})
        
        if risks.get("Cardiovascular Disease", 0) > 0.2:
            recommendations.append(matrix["Lycopene"])
        if risks.get("Diabetes", 0) > 0.2:
            recommendations.append(matrix["Sulforaphane"])
        if risks.get("Early Aging", 0) > 0.1:
            recommendations.append(matrix["Anthocyanin"])
            
        return recommendations

# =================================================================
# 3. GLOBAL INTERFACE (CLI for Demo)
# =================================================================

def run_english_demo():
    print("="*60)
    print("HealthBook MI Engine: Planetary Metabolism OS Demo")
    print("="*60)
    
    engine = MetabolicIntelligenceEngine()
    questions = MetabolicKnowledgeBase.get_diagnostic_questions()
    
    user_answers = {}
    print("\n[Diagnostic Interview Phase]")
    for q_id, item in questions.items():
        ans = input(f"Q{q_id}: {item['question']} (y/n): ").lower()
        user_answers[q_id] = True if ans == 'y' else False
    
    # Execution
    print("\n[Processing Metabolic Intelligence...]")
    report = engine.analyze_imbalance(user_answers)
    prescriptions = engine.prescribe_phytochemicals()
    
    # Output Results
    print("\n" + "!"*20 + " ANALYSIS REPORT " + "!"*20)
    print(f"Metabolic Threats Detected: {', '.join(report['metabolic_threats'])}")
    
    print("\nDisease Risk Profile:")
    for disease, score in report['disease_risk_profile'].items():
        intensity = "■" * int(score * 10)
        print(f" - {disease:25}: {intensity} ({score:.2f})")
        
    print("\nMetabolic Defense Prescription (Phytochemicals):")
    for p in prescriptions:
        print(f" + {p['source']} -> Extract: {p['function']} [{p['defense_layer']}]")
    print("="*60)

if __name__ == "__main__":
    run_english_demo()

```
