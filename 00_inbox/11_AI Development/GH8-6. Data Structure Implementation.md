### 2. The 137 Disease Correlation Matrix (`core/disease_engine.py`)

疾病と食生活、そして解決策（ファイトケミカル）を紐付ける推論コアです。

```python
"""
core/disease_engine.py
The engine that maps questionnaire vectors into 137 disease risks.
"""

class DiseaseRiskMatrix:
    def __init__(self):
        # Sample of the 137 Matrix
        self.matrix = {
            "Type_2_Diabetes": {
                "triggers": [1, 6, 25, 89], # Question IDs
                "pathway": "Glucose Homeostasis",
                "recommended_phytochemicals": ["Sulforaphane", "Quercetin"],
                "mbt_synergy": "Lactobacillus-Probiotics-A"
            },
            "Arteriosclerosis": {
                "triggers": [2, 5, 12, 110],
                "pathway": "Vascular Endothelium Inflammation",
                "recommended_phytochemicals": ["Lycopene", "EGCG"],
                "mbt_synergy": "Polyphenol-Active-MBT"
            }
        }

    def get_risk_score(self, disease_name, answered_ids):
        if disease_name not in self.matrix: return 0.0
        relevant_triggers = self.matrix[disease_name]["triggers"]
        matches = len(set(relevant_triggers) & set(answered_ids))
        return (matches / len(relevant_triggers)) if relevant_triggers else 0.0

```
