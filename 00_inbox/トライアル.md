import pandas as pd

# 1. マスターデータの定義
matrix_data = {
    "疾患名": ["大腸がん", "狭心症", "糖尿病", "胃がん", "眼精疲労"],
    "早食い": [1.0, 1.0, 1.0, 1.5, 0.0],
    "野菜不足": [2.0, 1.5, 1.0, 2.0, 2.0],
    "塩分過多": [1.0, 1.5, 1.0, 2.0, 0.0],
    "脂質過多": [2.0, 2.0, 1.5, 0.0, 0.0]
}
df_matrix = pd.DataFrame(matrix_data).set_index("疾患名")

def run_trial():
    print("--- BioNexus HealthBook 運用トライアル ---")
    user_answers = {}
    
    # 2. 動的な問診（ユーザー入力を受け付ける）
    for question in df_matrix.columns:
        ans = input(f"Q: {question} にあてはまりますか？ (y/n): ")
        user_answers[question] = 1 if ans.lower() == 'y' else 0

    # 3. 介入パラメータの入力
    print("\n[介入設定]")
    agrix = input("AGRIX産高品質野菜を摂取していますか？ (y/n): ")
    mbt = input("MBT55による腸内代謝は活性化していますか？ (y/n): ")

    agrix_boost = 0.5 if agrix.lower() == 'y' else 1.0
    mbt_efficiency = 1.2 if mbt.lower() == 'y' else 0.8

    # 4. 推論計算
    results = []
    for disease in df_matrix.index:
        base_score = sum(df_matrix.loc[disease][q] * val for q, val in user_answers.items())
        final_score = (base_score * agrix_boost) / mbt_efficiency
        results.append({"疾患": disease, "スコア": round(final_score, 2)})

    # 5. 診断レポート出力
    report = pd.DataFrame(results).sort_values(by="スコア", ascending=False)
    print("\n" + "="*30)
    print("【BioNexus 診断レポート】")
    print(report.to_string(index=False))
    print("="*30)
    
    top_disease = report.iloc[0]['疾患']
    print(f"\nアドバイス: 最優先リスクは【{top_disease}】です。")
    if agrix.lower() == 'n':
        print("→ AGRIXプラットフォーム指定の抗酸化作物の導入を推奨します。")

# 実行
run_trial()
