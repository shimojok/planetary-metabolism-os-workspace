## Part 1｜GraphRAG の詳細

### アーキテクチャ概要

```
従来のRAG:  Query → Vector Search → Top-K Chunks → LLM → Answer
                                                           ↑
                                               局所的な文脈しか見えない

GraphRAG:   Query → Entity Detection → Graph Traversal → Community Summary → LLM → Answer
                                              ↑
                                   グローバルな関係性を把握できる
```

### 2段階のインデックス構造

```python

# 各コミュニティの要約をLLMで生成し、階層的に保存
def generate_community_summaries(graph, llm):
    communities = graph.query("""
        MATCH (n)
        WITH n.communityId AS cid, collect(n) AS nodes
        RETURN cid, nodes LIMIT 100
    """)

    summaries = {}
    for community in communities:
        cid = community["cid"]
        nodes = community["nodes"]

        # コミュニティ内のノード・エッジを自然言語化
        context = build_community_context(nodes, graph)

        summary = llm.invoke(f"""
            以下のエンティティ群が形成するコミュニティの要約を200字以内で生成してください。
            主要テーマ、重要なエンティティ、関係性のパターンを含めること。
            
            コンテキスト: {context}
        """)
        summaries[cid] = summary.content

    return summaries

# ===== クエリ時: グローバル + ローカル検索の融合 =====

def graphrag_query(question: str, graph, summaries, vector_store):
    # 1. グローバル検索: コミュニティサマリーをベクトル検索
    global_results = vector_store.similarity_search(
        question, 
        filter={"type": "community_summary"},
        k=5
    )

    # 2. ローカル検索: エンティティからグラフ探索
    entities = extract_entities(question, llm)
    local_results = graph.query(f"""
        MATCH (e {{name: $entity}})-[r*1..3]-(related)
        RETURN e, r, related
        LIMIT 50
    """, {"entity": entities[0]})

    # 3. 融合して最終回答生成
    context = merge_results(global_results, local_results)
    return llm.invoke(f"質問: {question}\n\nコンテキスト: {context}")
```

## Part 2｜LangGraph 実装例（Agentic KG の中核）

### エージェントグラフの全体設計

```python
from langgraph.graph import StateGraph, END
from langgraph.prebuilt import ToolNode
from typing import TypedDict, Annotated
import operator

# ===== State 定義 =====
class KGAgentState(TypedDict):
    query: str
    entities: list[dict]          # 抽出されたエンティティ
    graph_context: str            # KGから取得した文脈
    reasoning_steps: Annotated[list, operator.add]  # 推論ステップ蓄積
    new_triples: list[dict]       # 新規追加トリプル
    final_answer: str
    confidence: float

# ===== ノード定義 =====

def entity_extraction_node(state: KGAgentState) -> KGAgentState:
    """クエリからエンティティを抽出"""
    prompt = f"""
    以下のクエリからエンティティと関係を抽出してください。
    JSON形式で返してください: {{"entities": [...], "relations": [...]}}
    
    クエリ: {state['query']}
    """
    result = llm.invoke(prompt)
    parsed = json.loads(result.content)
    
    return {**state, 
            "entities": parsed["entities"],
            "reasoning_steps": [f"エンティティ抽出: {parsed['entities']}"]}

def graph_retrieval_node(state: KGAgentState) -> KGAgentState:
    """Knowledge Graphからサブグラフを取得"""
    entity_names = [e["name"] for e in state["entities"]]
    
    cypher = """
    MATCH path = (start)-[*1..3]-(end)
    WHERE start.name IN $entities
    WITH path, 
         reduce(score=0, r IN relationships(path) | score + r.weight) AS path_score
    ORDER BY path_score DESC
    RETURN path LIMIT 30
    """
    
    subgraph = graph.query(cypher, {"entities": entity_names})
    context = serialize_subgraph(subgraph)
    
    return {**state,
            "graph_context": context,
            "reasoning_steps": [f"グラフ取得: {len(subgraph)}パス"]}

def reasoning_node(state: KGAgentState) -> KGAgentState:
    """取得した文脈をもとに推論"""
    prompt = f"""
    あなたはKnowledge Graphの推論エージェントです。
    
    クエリ: {state['query']}
    グラフ文脈: {state['graph_context']}
    これまでの推論: {state['reasoning_steps']}
    
    Step-by-stepで推論し、以下を返してください:
    1. 回答
    2. 確信度 (0.0-1.0)
    3. グラフに追加すべき新規トリプル（あれば）
    """
    result = llm.invoke(prompt)
    parsed = parse_reasoning_result(result.content)
    
    return {**state,
            "final_answer": parsed["answer"],
            "confidence": parsed["confidence"],
            "new_triples": parsed.get("new_triples", []),
            "reasoning_steps": [f"推論完了: 確信度={parsed['confidence']}"]}

def graph_update_node(state: KGAgentState) -> KGAgentState:
    """新規知識をグラフに書き込む（自己組織化）"""
    for triple in state["new_triples"]:
        graph.query("""
            MERGE (s:Entity {name: $subject})
            MERGE (o:Entity {name: $object})
            MERGE (s)-[r:RELATION {type: $predicate}]->(o)
            SET r.confidence = $confidence,
                r.created_at = timestamp(),
                r.source = 'agent_inference'
        """, {
            "subject": triple["subject"],
            "predicate": triple["predicate"],
            "object": triple["object"],
            "confidence": state["confidence"]
        })
    
    return state

def should_update_graph(state: KGAgentState) -> str:
    """条件分岐: 確信度が高い場合のみグラフ更新"""
    if state["confidence"] >= 0.85 and state["new_triples"]:
        return "update"
    return "skip"

# ===== グラフ組み立て =====

workflow = StateGraph(KGAgentState)

workflow.add_node("extract_entities", entity_extraction_node)
workflow.add_node("retrieve_graph",   graph_retrieval_node)
workflow.add_node("reason",           reasoning_node)
workflow.add_node("update_graph",     graph_update_node)

workflow.set_entry_point("extract_entities")
workflow.add_edge("extract_entities", "retrieve_graph")
workflow.add_edge("retrieve_graph",   "reason")

# 条件付きエッジ（自己組織化の核心）
workflow.add_conditional_edges(
    "reason",
    should_update_graph,
    {"update": "update_graph", "skip": END}
)
workflow.add_edge("update_graph", END)

agent = workflow.compile()

# ===== 実行 =====
result = agent.invoke({
    "query": "新薬XとタンパクYの相互作用は？",
    "entities": [], "graph_context": "",
    "reasoning_steps": [], "new_triples": [],
    "final_answer": "", "confidence": 0.0
})
print(result["final_answer"])
```

## Part 3｜自己組織化メカニズムの実装

### 4つの自己組織化プロセス

```python
import numpy as np
from sklearn.cluster import HDBSCAN
from sentence_transformers import SentenceTransformer

encoder = SentenceTransformer("intfloat/multilingual-e5-large")

# ========================================
# 1. 動的クラスタリング（スキーマ自動生成）
# ========================================

class DynamicSchemaOrganizer:
    def __init__(self, graph, threshold=0.75):
        self.graph = graph
        self.threshold = threshold

    def auto_cluster_new_entities(self, new_entities: list[str]):
        # 埋め込みベクトルで意味的クラスタリング
        embeddings = encoder.encode(new_entities)
        
        clusterer = HDBSCAN(
            min_cluster_size=3,
            metric='cosine',
            cluster_selection_epsilon=0.15
        )
        labels = clusterer.fit_predict(embeddings)
        
        for cluster_id in set(labels):
            if cluster_id == -1:  # ノイズ点はスキップ
                continue
            
            cluster_members = [
                new_entities[i] for i, l in enumerate(labels) 
                if l == cluster_id
            ]
            
            # クラスターの意味をLLMで命名
            type_name = self._name_cluster(cluster_members)
            
            # 新しいノード型をスキーマに追加
            self.graph.query(f"""
                MATCH (n) WHERE n.name IN $members
                SET n:{type_name}
            """, {"members": cluster_members})
            
            print(f"✅ 新型 '{type_name}' を自動生成: {cluster_members}")

    def _name_cluster(self, members: list[str]) -> str:
        result = llm.invoke(
            f"これらのエンティティに共通する概念型を1語（英語）で答えてください: {members}"
        )
        return result.content.strip().replace(" ", "_")


# ========================================
# 2. 重み付き忘却（Weighted Forgetting）
# ========================================

class AdaptiveMemoryManager:
    """使用頻度・鮮度・重要度でエッジを動的に管理"""

    def decay_and_prune(self):
        # 時間減衰 + 使用頻度で重みを更新
        self.graph.query("""
            MATCH ()-[r]-()
            WHERE r.source = 'agent_inference'
            SET r.weight = r.weight 
                * exp(-0.01 * (timestamp() - r.last_accessed) / 86400000)
                -- 日単位で指数減衰
        """)

        # 閾値以下のエッジを削除（プルーニング）
        pruned = self.graph.query("""
            MATCH ()-[r]-()
            WHERE r.weight < 0.05 AND r.source = 'agent_inference'
            DELETE r
            RETURN count(r) AS pruned_count
        """)
        print(f"🗑️ プルーニング: {pruned[0]['pruned_count']}エッジ削除")

    def reinforce_accessed_path(self, path_nodes: list[str]):
        # アクセスされたパスの重みを強化
        self.graph.query("""
            MATCH (a)-[r]-(b)
            WHERE a.name IN $nodes AND b.name IN $nodes
            SET r.weight = min(1.0, r.weight * 1.2),
                r.last_accessed = timestamp()
        """, {"nodes": path_nodes})


# ========================================
# 3. 矛盾検出と自動解決
# ========================================

class ContradictionResolver:
    def detect_and_resolve(self):
        # 同一エンティティペア間の矛盾関係を検出
        contradictions = self.graph.query("""
            MATCH (a)-[r1:RELATION]->(b), (a)-[r2:RELATION]->(b)
            WHERE r1.type <> r2.type
              AND r1.confidence > 0.7 AND r2.confidence > 0.7
            RETURN a.name, b.name, r1.type, r2.type,
                   r1.confidence, r2.confidence
        """)

        for c in contradictions:
            resolution = llm.invoke(f"""
                知識グラフで矛盾を検出しました:
                - ({c['a.name']}) -[{c['r1.type']}]-> ({c['b.name']})  信頼度: {c['r1.confidence']}
                - ({c['a.name']}) -[{c['r2.type']}]-> ({c['b.name']})  信頼度: {c['r2.confidence']}
                
                どちらが正しいか、または両方を統合する新しい関係を提案してください。
                JSON: {{"action": "keep_r1|keep_r2|merge", "merged_type": "...", "reason": "..."}}
            """)
            self._apply_resolution(c, json.loads(resolution.content))


# ========================================
# 4. スキーマバージョン管理
# ========================================

class SchemaVersionController:
    def commit_schema_change(self, change_description: str):
        version = self.get_current_version() + 1
        self.graph.query("""
            CREATE (v:SchemaVersion {
                version: $version,
                description: $desc,
                timestamp: timestamp(),
                snapshot: $snapshot
            })
        """, {
            "version": version,
            "desc": change_description,
            "snapshot": self.export_schema()
        })
```

```python
# 薬剤再利用候補の自動発見
def find_drug_repurposing_candidates(target_disease: str):
    result = graph.query("""
        MATCH (drug:Drug)-[:INHIBITS|ACTIVATES]->(protein:Protein)
              -[:INVOLVED_IN]->(pathway:Pathway)
              -[:DYSREGULATED_IN]->(disease:Disease)
        WHERE disease.name = $disease
          AND NOT (drug)-[:APPROVED_FOR]->(disease)
        WITH drug, count(DISTINCT pathway) AS pathway_count,
             avg(r.confidence) AS avg_confidence
        WHERE pathway_count >= 2
        RETURN drug.name, pathway_count, avg_confidence
        ORDER BY pathway_count DESC, avg_confidence DESC
        LIMIT 10
    """, {"disease": target_disease})
    return result
```

```python
# システミックリスクのグラフ分析
def calculate_systemic_risk(entity: str, depth: int = 3):
    # Pagerank的なリスク伝播スコア計算
    risk_score = graph.query("""
        CALL gds.pageRank.stream('financial-kg', {
            maxIterations: 20,
            dampingFactor: 0.85,
            relationshipWeightProperty: 'exposure_amount'
        })
        YIELD nodeId, score
        WITH gds.util.asNode(nodeId) AS node, score
        WHERE node.name = $entity
        RETURN score AS systemic_importance
    """, {"entity": entity})
    
    # 連鎖破綻シミュレーション（グラフ探索）
    cascade = graph.query("""
        MATCH path = (trigger {name: $entity})-[*1..$depth]-(affected)
        WHERE ALL(r IN relationships(path) WHERE r.exposure > 1000000)
        RETURN affected.name, length(path) AS distance,
               reduce(risk=1.0, r IN relationships(path) | risk * r.contagion_rate) AS cascade_risk
        ORDER BY cascade_risk DESC
    """, {"entity": entity, "depth": depth})
    
    return {"systemic_importance": risk_score, "cascade_analysis": cascade}
```

```python
# 攻撃パターンのグラフマッチング（MITRE ATT&CK フレームワーク統合）
def detect_apt_pattern(observed_events: list[dict]):
    # 観測イベントをKGに一時ノードとして追加
    graph.query("""
        UNWIND $events AS event
        MERGE (e:ObservedEvent {id: event.id})
        SET e += event
    """, {"events": observed_events})

    # MITRE ATT&CK の既知APTパターンとサブグラフマッチング
    matches = graph.query("""
        MATCH (apt:APTGroup)-[:USES]->(technique:Technique)
              -[:OBSERVED_AS]->(obs:ObservedEvent)
        WITH apt, count(DISTINCT technique) AS matched_techniques,
             apt.total_techniques AS total
        WHERE toFloat(matched_techniques) / total > 0.6
        RETURN apt.name, matched_techniques, total,
               toFloat(matched_techniques)/total AS match_ratio
        ORDER BY match_ratio DESC
    """)
    return matches
```
