def extract_nodes_edges(chunk, client):
    prompt = f"""
    You are an ontology extractor for MBT55 system.

    Extract in JSON:

    {{
      "nodes": [
        {{"name": "", "type": "Microbe | Metabolite | ..."}}
      ],
      "edges": [

      ]
    }}

        model="gpt-4o-mini",

## ■ ノード保存

```python
def save_to_neo4j(nodes, edges):
    with driver.session() as session:
        for n in nodes:
            session.run(
                "MERGE (a:Entity {name: $name, type: $type})",
                name=n["name"], type=n["type"]
            )

        for e in edges:
            session.run(
                """
                MATCH (a {name: $source}), (b {name: $target})
                MERGE (a)-[:REL {type: $rel}]->(b)
                """,
                source=e["source"],
                target=e["target"],
                rel=e["relation"]
            )
```
