```python
# extractor.py

import json
from openai import OpenAI
from normalizer import normalize_entity

client = OpenAI()

def extract_nodes_edges(chunk: str):

    prompt = f"""
You are an MBT55 ontology extractor.

Return ONLY JSON:

{{
  "nodes": [
    {{"name": "", "type": ""}}
  ],
  "edges": [
    {{"source": "", "target": "", "relation": ""}}
  ]
}}

Rules:
- Use scientific naming when possible
- Keep consistency
- No explanation

Text:
{chunk}
"""

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
        response_format={"type": "json_object"}
    )

    data = json.loads(response.choices[0].message.content)

    # 正規化処理
    for node in data["nodes"]:
        node["name"] = normalize_entity(node["name"])

    for edge in data["edges"]:
        edge["source"] = normalize_entity(edge["source"])
        edge["target"] = normalize_entity(edge["target"])

    return data
```

# ■ STEP 6：graph_builder.py 修正

```python
# graph_builder.py

import networkx as nx

G = nx.DiGraph()

def add_to_graph(data):
    nodes = data["nodes"]
    edges = data["edges"]

    for n in nodes:
        G.add_node(n["name"], type=n["type"])

    for e in edges:
        G.add_edge(
            e["source"],
            e["target"],
            relation=e["relation"]
        )


def save_graph(path):
    nx.write_gml(G, path)
```

```python
def split_document(doc, config):

    prompt = f"""
Split into sections (max {config['max_splits']}).

Return JSON:
{{
  "chunks": ["text1", "text2"]
}}

Text:
{doc}
"""

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
        response_format={"type": "json_object"}
    )

    data = json.loads(response.choices[0].message.content)
    return data["chunks"]
```
