## 1.2 分割ツールの作成（これが核心）

`splitter.py` というファイルを作成し、以下をコピー＆ペースト：

```python
import os
import json
from pathlib import Path
from typing import List, Dict
import openai
from dotenv import load_dotenv

load_dotenv()
client = openai.OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

# 分割設定（Kaz様が決定する部分）
SPLIT_CONFIG = {
    "max_splits": 12,           # 1ノート最大12分割
    "min_chunk_size": 500,      # 最小500文字
    "target_folders": {          # 分類先フォルダ
        "MBT55": "10_Knowledge/MBT55",
        "AGRIX": "10_Knowledge/AGRIX",
        "AgriWare": "10_Knowledge/AgriWare",
        "HealthBook": "10_Knowledge/HealthBook",
        "Microsoft": "20_Projects/Microsoft",
        "Gates": "20_Projects/Gates_Foundation",
        "Rockefeller": "20_Projects/Rockefeller",
        "WorldBank": "20_Projects/World_Bank",
        "Climate": "10_Knowledge/Climate",
        "Code": "40_Code/Python",
        "JSON": "40_Code/JSON",
        "default": "10_Knowledge/General"
    }
}

def split_note_by_ai(file_path: str) -> List[Dict]:
    """1つの.mdファイルをAIが最適な粒度に分割"""
    
    with open(file_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    prompt = f"""
    以下のObsidianノートを、AIエージェントが理解しやすい単位に分割してください。
    
    分割ルール：
    - 最大{SPLIT_CONFIG['max_splits']}分割
    - 各チャンクは{SPLIT_CONFIG['min_chunk_size']}文字以上
    - 意味的なまとまりを維持
    - 見出し（#）がある場合はそこで分割を優先
    
    出力は以下のJSON形式で：
    {{
        "chunks": [
            {{
                "title": "チャンクのタイトル",
                "content": "分割された内容",
                "category": "MBT55/AGRIX/HealthBook/Code/JSON/Generalから選択",
                "keywords": ["キーワード1", "キーワード2"]
            }}
        ]
    }}
    
    ノート内容：
    {content[:15000]}  # 最初の15000文字（必要に応じて調整）
    """
    
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
        response_format={"type": "json_object"}
    )
    
    result = json.loads(response.choices[0].message.content)
    return result["chunks"]

def save_split_chunks(chunks: List[Dict], original_filename: str):
    """分割されたチャンクを適切なフォルダに保存"""
    
    for i, chunk in enumerate(chunks):
        # カテゴリに応じた保存先フォルダを決定
        category = chunk.get("category", "default")
        folder = SPLIT_CONFIG["target_folders"].get(category, SPLIT_CONFIG["target_folders"]["default"])
        
        # フォルダがなければ作成
        Path(folder).mkdir(parents=True, exist_ok=True)
        
        # ファイル名を生成（元ファイル名 + 番号）
        base_name = Path(original_filename).stem
        new_filename = f"{folder}/{base_name}_{i+1:02d}_{chunk['title'][:30]}.md"
        
        # 内容を保存
        with open(new_filename, 'w', encoding='utf-8') as f:
            f.write(f"# {chunk['title']}\n\n")
            f.write(f"*元ファイル: {original_filename}*\n\n")
            f.write(chunk['content'])
        
        print(f"✅ 保存: {new_filename}")

def process_vault(vault_path: str):
    """Vault内の全.mdファイルを処理"""
    
    md_files = list(Path(vault_path).rglob("*.md"))
    print(f"📊 合計 {len(md_files)} 個の.mdファイルを発見")
    
    for i, file_path in enumerate(md_files):
        print(f"\n📄 処理中 [{i+1}/{len(md_files)}]: {file_path}")
        
        try:
            chunks = split_note_by_ai(str(file_path))
            save_split_chunks(chunks, str(file_path))
        except Exception as e:
            print(f"❌ エラー: {e}")
            continue

if __name__ == "__main__":
    # 実行
    process_vault("./vault_original")
```

import json
from pathlib import Path
import networkx as nx
from typing import List, Dict
import openai
from dotenv import load_dotenv
import os

load_dotenv()
client = openai.OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

def extract_entities_from_md(file_path: str) -> Dict:
    """1つの.mdファイルからエンティティと関係を抽出"""
    
    with open(file_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    prompt = f"""
    以下のテキストから、Planetary Metabolic OSの知識グラフに含めるべき
    エンティティ（ノード）とその関係（エッジ）を抽出してください。
    
    エンティティの種類：
    - Microbe（微生物）
    - Metabolite（代謝物）
    - Process（プロセス）
    - System（システム）
    - Technology（技術）
    - Organization（組織）
    - Person（人物）
    - Location（場所）
    - Concept（概念）
    
    出力JSON形式：
    {{
        "entities": [
            {{"name": "エンティティ名", "type": "種類", "description": "説明"}}
        ],
        "relations": [
            {{"source": "元エンティティ", "target": "先エンティティ", "relation": "関係の種類"}}
        ]
    }}
    
    テキスト：
    {content[:10000]}
    """
    
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
        response_format={"type": "json_object"}
    )
    
    return json.loads(response.choices[0].message.content)

def build_graph_from_vault(vault_path: str):
    """Vault内の全.mdから知識グラフを構築"""
    
    G = nx.DiGraph()
    md_files = list(Path(vault_path).rglob("*.md"))
    
    for file_path in md_files:
        print(f"📄 解析中: {file_path}")
        
        try:
            data = extract_entities_from_md(str(file_path))
            
            # エンティティ追加
            for entity in data.get("entities", []):
                G.add_node(
                    entity["name"],
                    type=entity["type"],
                    description=entity.get("description", ""),
                    source_file=str(file_path)
                )
            
            # 関係追加
            for rel in data.get("relations", []):
                G.add_edge(
                    rel["source"],
                    rel["target"],
                    relation=rel["relation"],
                    source_file=str(file_path)
                )
                
        except Exception as e:
            print(f"❌ エラー: {e}")
            continue
    
    # グラフを保存
    nx.write_gexf(G, "pmos_knowledge_graph.gexf")
    print(f"\n✅ グラフ構築完了: {G.number_of_nodes()} ノード, {G.number_of_edges()} エッジ")
    
    return G

if __name__ == "__main__":
    G = build_graph_from_vault("./vault_processed")

# フェーズ3：LangGraphで最初のAIエージェントを作る

## 3.1 最初のエージェント：MBT55科学エージェント

`mbt55_science_agent.py` を作成：

```python
from langgraph.graph import StateGraph, MessageGraph
from typing import TypedDict, List
import openai
from dotenv import load_dotenv
import os
from pathlib import Path

load_dotenv()
client = openai.OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

# エージェントの状態定義
class AgentState(TypedDict):
    query: str
    context: List[str]
    response: str
    sources: List[str]

# ナレッジベースから関連情報を取得
def retrieve_knowledge(state: AgentState) -> AgentState:
    """MBT55関連の知識をVaultから取得"""
    
    mbt55_files = list(Path("./10_Knowledge/MBT55").rglob("*.md"))
    relevant_chunks = []
    
    # 簡易版：キーワードマッチ（本番はベクトル検索に）
    keywords = state["query"].lower().split()
    
    for file_path in mbt55_files[:5]:  # 最初の5ファイル
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()
            if any(k in content.lower() for k in keywords):
                relevant_chunks.append(content[:1000])
                state["sources"].append(str(file_path))
    
    state["context"] = relevant_chunks
    return state

# LLMで回答生成
def generate_response(state: AgentState) -> AgentState:
    """コンテキストを元に回答生成"""
    
    prompt = f"""
    あなたはMBT55の科学メカニズムに特化したAIエージェントです。
    以下のコンテキストを元に、ユーザーの質問に答えてください。
    
    コンテキスト：
    {' '.join(state['context'])}
    
    ユーザーの質問：{state['query']}
    
    回答は科学的に正確で、必要に応じて数式やメカニズムの説明を含めてください。
    """
    
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}]
    )
    
    state["response"] = response.choices[0].message.content
    return state

# グラフの構築
def build_mbt55_agent():
    workflow = StateGraph(AgentState)
    
    workflow.add_node("retrieve", retrieve_knowledge)
    workflow.add_node("generate", generate_response)
    
    workflow.set_entry_point("retrieve")
    workflow.add_edge("retrieve", "generate")
    workflow.set_finish_point("generate")
    
    return workflow.compile()

# エージェントの実行
def ask_mbt55(query: str):
    agent = build_mbt55_agent()
    result = agent.invoke({
        "query": query,
        "context": [],
        "response": "",
        "sources": []
    })
    
    print(f"🤖 回答：\n{result['response']}\n")
    print(f"📚 参照元：")
    for src in result['sources']:
        print(f"  - {src}")

# 使用例
if __name__ == "__main__":
    ask_mbt55("MBT55のメタン削減メカニズムを説明してください")
```

    Path(path).parent.mkdir(parents=True, exist_ok=True)
    with open(path, 'w', encoding='utf-8') as f:
        f.write(content)

print("✅ GitHub用フォルダ構造を作成しました")

# フェーズ5：ビル・ゲイツ氏・Microsoft向け提案の自動生成

## 5.1 提案書生成エージェント

`gates_proposal_agent.py`：

```python
from langgraph.graph import StateGraph
from typing import TypedDict, List
import openai
from pathlib import Path

class ProposalState(TypedDict):
    sections: List[str]
    content: str
    final_proposal: str

def gather_evidence(state: ProposalState) -> ProposalState:
    """MBT55/AGRIX/HealthBookの実績データを収集"""
    
    evidence_files = [
        "./10_Knowledge/MBT55/performance_data.md",
        "./10_Knowledge/AGRIX/ghg_reduction.md",
        "./10_Knowledge/HealthBook/clinical_data.md",
        "./20_Projects/Gates_Foundation/previous_communications.md"
    ]
    
    evidence = []
    for file in evidence_files:
        if Path(file).exists():
            with open(file, 'r', encoding='utf-8') as f:
                evidence.append(f.read())
    
    state["sections"] = evidence
    return state

def write_proposal(state: ProposalState) -> ProposalState:
    """ゲイツ財団向け提案書を生成"""
    
    prompt = f"""
    以下の情報を元に、ビル・ゲイツ氏とビル・アンド・メリンダ・ゲイツ財団向けの
    提案書を作成してください。
    
    提案内容：
    - MBT55, AGRIX, HealthBookの3事業をMicrosoftに有償譲渡
    - 世界の食料問題、温室効果ガス削減、劣化土壌修復、グローバルヘルスへの貢献
    
    裏付けデータ：
    {' '.join(state['sections'])}
    
    提案書の構成：
    1. エグゼクティブサマリー
    2. 3事業の概要と独自性
    3. 科学的メカニズム（MBT55のteleodynamics）
    4. 定量化されたインパクト（GHG削減量、土壌修復面積、健康改善効果）
    5. Microsoftとのシナジー
    6. 譲渡条件と次のステップ
    
    形式：ビジネス文書形式、英語
    """
    
    response = openai.OpenAI().chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": prompt}]
    )
    
    state["final_proposal"] = response.choices[0].message.content
    return state

# 提案書生成
def generate_gates_proposal():
    workflow = StateGraph(ProposalState)
    workflow.add_node("gather", gather_evidence)
    workflow.add_node("write", write_proposal)
    workflow.set_entry_point("gather")
    workflow.add_edge("gather", "write")
    workflow.set_finish_point("write")
    
    agent = workflow.compile()
    result = agent.invoke({"sections": [], "content": "", "final_proposal": ""})
    
    # 保存
    with open("./targets/gates_foundation/proposal_final.md", 'w', encoding='utf-8') as f:
        f.write(result["final_proposal"])
    
    print("✅ 提案書を生成しました")

if __name__ == "__main__":
    generate_gates_proposal()
```
