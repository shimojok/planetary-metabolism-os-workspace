```python
"""
超高速探索エージェント - ルールベース＋軽量AIのハイブリッド
"""

from pathlib import Path
import json
import re

class FastExplorer:
    def __init__(self, vault_path: str):
        self.vault_path = Path(vault_path)
        
    def fast_scan(self) -> list:
        """ルールベースで重要ノートを特定（AI呼び出し最小化）"""
        important_keywords = [
            'mbt55', 'ghg', 'co2', 'メタン', 'n2o', '削減',
            'ゲイツ', 'gates', 'microsoft', 'azure',
            'シミュレーション', '数式', 'データ', '実証',
            'teleodynamics', 'ハイパーサイクル'
        ]
        
        priority_notes = []
        
        for md in self.vault_path.rglob("*.md"):
            # ファイルサイズと更新日でフィルタ
            if md.stat().st_size < 1000:  # 1000バイト未満はスキップ
                continue
                
            # ファイル名で重要度判定
            score = 0
            name_lower = md.name.lower()
            for kw in important_keywords:
                if kw in name_lower:
                    score += 2
            
            # 内容の冒頭をチェック
            try:
                with open(md, 'r', encoding='utf-8') as f:
                    first_500 = f.read(500).lower()
                    for kw in important_keywords:
                        if kw in first_500:
                            score += 1
            except:
                pass
            
            if score >= 3:  # スコア3以上を優先
                priority_notes.append({
                    "path": str(md),
                    "score": score,
                    "size": md.stat().st_size
                })
        
        # スコア順にソート
        priority_notes.sort(key=lambda x: x['score'], reverse=True)
        
        # 上位100件を保存
        with open('data/priority_fast.json', 'w') as f:
            json.dump(priority_notes[:100], f, indent=2)
        
        return priority_notes[:100]

# 実行（3時間）
explorer = FastExplorer("../vault_original")
priority = explorer.fast_scan()
print(f"✅ {len(priority)} 件の優先ノートを特定")
```

```python
"""
高速グラフ構築 - バッチ処理で一気に投入
"""

from neo4j import GraphDatabase
import json
from pathlib import Path

class FastGraphBuilder:
    def __init__(self):
        self.driver = GraphDatabase.driver(
            "bolt://localhost:7687",
            auth=("neo4j", "password")
        )
    
    def batch_insert_nodes(self, notes: list):
        """バッチインサートで高速化"""
        with self.driver.session() as session:
            # 一括作成用クエリ
            for note in notes[:20]:  # 20件ずつバッチ
                session.run("""
                    CREATE (n:Note {
                        path: $path,
                        name: $name,
                        score: $score
                    })
                    """,
                    path=note['path'],
                    name=Path(note['path']).stem,
                    score=note.get('score', 0)
                )
    
    def create_indexes(self):
        """インデックス作成で検索高速化"""
        with self.driver.session() as session:
            session.run("CREATE INDEX note_path IF NOT EXISTS FOR (n:Note) ON (n.path)")
            session.run("CREATE INDEX note_name IF NOT EXISTS FOR (n:Note) ON (n.name)")

# 実行
builder = FastGraphBuilder()
with open('data/priority_fast.json') as f:
    notes = json.load(f)
builder.batch_insert_nodes(notes)
builder.create_indexes()
```

```python
"""
MVPシミュレーター - ゲイツ向け数値を最優先
"""

import numpy as np
from scipy.integrate import odeint

class HypercycleMVP:
    def __init__(self):
        self.params = {
            'base_growth': 0.2,
            'mbt_effect': 1.5,    # MBT55投入効果
            'agrix_effect': 1.3,   # AGRIX政策効果
            'health_effect': 1.2,   # HealthBook効果
            'decay': 0.1
        }
    
    def calculate_ghg_impact(self, years=10, hectares=1e6):
        """GHG削減量を一発計算（ゲイツ向け）"""
        
        # ベースライン
        base_ghg = 5.8  # tCO2e/ha/年（既存データ）
        
        # 統合効果
        total_effect = (
            self.params['mbt_effect'] *
            self.params['agrix_effect'] *
            self.params['health_effect']
        )
        
        annual_reduction = base_ghg * total_effect * hectares
        
        return {
            "annual_reduction_tCO2e": annual_reduction,
            "ten_year_total": annual_reduction * years,
            "hectares_covered": hectares,
            "co2_equivalent_cars": annual_reduction / 4.6,  # 乗用車換算
            "roi_estimate": f"{total_effect*100-100:.0f}% improvement"
        }
```

```python
"""
高速分類器 - 並列処理で一気に分類
"""

from concurrent.futures import ThreadPoolExecutor, as_completed
import openai
import json

class FastClassifier:
    def __init__(self):
        self.client = openai.OpenAI()
        
    def classify_batch(self, notes: list, batch_size=10):
        """バッチ処理で高速分類"""
        
        results = []
        with ThreadPoolExecutor(max_workers=5) as executor:
            futures = {
                executor.submit(self._classify_one, note): note
                for note in notes[:50]  # 最初の50件
            }
            
            for future in as_completed(futures):
                result = future.result()
                results.append(result)
                print(f"✅ {len(results)}/50完了")
        
        return results
    
    def _classify_one(self, note):
        """1件分類"""
        with open(note['path'], 'r') as f:
            content = f.read()[:1000]
        
        prompt = f"Classify as MBT55/AGRIX/HealthBook/Other. Return JSON: {{'domain':'','confidence':0}}"
        
        response = self.client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[{"role": "user", "content": prompt + "\n\n" + content}],
            response_format={"type": "json_object"}
        )
        
        return {
            "path": note['path'],
            "classification": json.loads(response.choices[0].message.content)
        }
```

```python
"""
軽量スーパーバイザー - ルールベース＋LLM
"""

class FastSupervisor:
    def __init__(self):
        self.agents = {}
        
    def route(self, query):
        """キーワードベースでルーティング"""
        query_lower = query.lower()
        
        if 'mbt55' in query_lower or '微生物' in query_lower:
            return 'mbt55'
        elif 'agrix' in query_lower or '農業' in query_lower:
            return 'agrix'
        elif 'health' in query_lower or '健康' in query_lower:
            return 'healthbook'
        else:
            return 'general'
    
    def orchestrate(self, query):
        agent_name = self.route(query)
        if agent_name in self.agents:
            return self.agents[agent_name].process(query)
        return {"result": "General response"}
```
