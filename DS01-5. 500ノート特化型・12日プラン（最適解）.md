```python
"""
核心500ノート抽出エージェント
"""

from pathlib import Path
import json
import re

class CoreExtractor:
    def __init__(self, vault_path: str):
        self.vault_path = Path(vault_path)
        
        # ゲイツ提案に必須のキーワード（重み付き）
        self.keywords = {
            # 最重要（スコア5）
            'mbt55': 5, 'ghg': 5, 'co2': 5, 'メタン': 5, 'n2o': 5,
            'ゲイツ': 5, 'gates': 5, 'microsoft': 5, 'azure': 5,
            'teleodynamics': 5, 'ハイパーサイクル': 5,
            
            # 重要（スコア3）
            'シミュレーション': 3, '数式': 3, 'データ': 3,
            '実証': 3, '削減量': 3, '収量': 3, 'コスト': 3,
            'agrix': 3, 'healthbook': 3, 'pbpe': 3,
            
            # 関連（スコア1）
            '土壌': 1, '微生物': 1, '発酵': 1, '肥料': 1,
            '健康': 1, '腸内': 1, '代謝': 1, '政策': 1
        }
    
    def score_note(self, file_path: Path) -> int:
        """ノートのスコアリング"""
        score = 0
        name_lower = file_path.name.lower()
        
        # ファイル名でのスコア
        for kw, weight in self.keywords.items():
            if kw in name_lower:
                score += weight * 2  # ファイル名は重視
        
        # 内容の先頭2000文字でスコア
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                content = f.read(2000).lower()
                
            for kw, weight in self.keywords.items():
                if kw in content:
                    score += weight
                    
                    # 複数回出現でボーナス
                    count = content.count(kw)
                    if count > 3:
                        score += weight // 2
        except:
            pass
        
        return score
    
    def extract_core_notes(self, limit: int = 500):
        """核心ノートを抽出"""
        all_notes = []
        
        for md in self.vault_path.rglob("*.md"):
            # 最低サイズ（1KB以上）
            if md.stat().st_size < 1000:
                continue
                
            score = self.score_note(md)
            if score >= 10:  # スコア10以上のみ
                all_notes.append({
                    "path": str(md),
                    "score": score,
                    "size": md.stat().st_size,
                    "name": md.name
                })
        
        # スコア順にソート
        all_notes.sort(key=lambda x: x['score'], reverse=True)
        
        # 上位500件を保存
        core_notes = all_notes[:limit]
        
        # 内訳分析
        domains = {'MBT55': 0, 'AGRIX': 0, 'HealthBook': 0, 'Other': 0}
        for note in core_notes:
            path_lower = note['path'].lower()
            if 'mbt55' in path_lower:
                domains['MBT55'] += 1
            elif 'agrix' in path_lower:
                domains['AGRIX'] += 1
            elif 'healthbook' in path_lower:
                domains['HealthBook'] += 1
            else:
                domains['Other'] += 1
        
        # 保存
        result = {
            "total_notes": len(all_notes),
            "core_notes": core_notes,
            "domain_distribution": domains
        }
        
        with open('data/core_500.json', 'w', encoding='utf-8') as f:
            json.dump(result, f, indent=2, ensure_ascii=False)
        
        print(f"✅ 核心500ノート抽出完了")
        print(f"内訳: {domains}")
        
        return core_notes

# 実行
extractor = CoreExtractor("../vault_original")
core_notes = extractor.extract_core_notes(500)
```

```python
import json
from collections import Counter
import re

class CoreAnalyzer:
    def __init__(self):
        with open('data/core_500.json') as f:
            self.data = json.load(f)
        
    def extract_metadata(self):
        """メタデータ抽出"""
        notes = self.data['core_notes']
        
        # 更新日分布
        import os
        from datetime import datetime
        
        dates = []
        sizes = []
        for note in notes:
            stat = os.stat(note['path'])
            dates.append(datetime.fromtimestamp(stat.st_mtime).strftime('%Y-%m'))
            sizes.append(stat.st_size)
        
        return {
            "total_size_mb": sum(sizes) / (1024*1024),
            "avg_size_kb": (sum(sizes)/len(sizes))/1024,
            "recent_months": Counter(dates).most_common(5),
            "score_range": {
                "min": min(n['score'] for n in notes),
                "max": max(n['score'] for n in notes),
                "avg": sum(n['score'] for n in notes)/len(notes)
            }
        }

analyzer = CoreAnalyzer()
```

```python
"""
核心ノート分類（GPT-4o-mini一括処理）
"""

import openai
import json
from concurrent.futures import ThreadPoolExecutor, as_completed

class CoreClassifier:
    def __init__(self):
        self.client = openai.OpenAI()
        
    def classify_batch(self, notes, batch_size=20):
        """20件ずつバッチ分類"""
        
        results = []
        total = len(notes)
        
        for i in range(0, total, batch_size):
            batch = notes[i:i+batch_size]
            print(f"📊 バッチ {i//batch_size + 1}/{(total-1)//batch_size + 1}")
            
            with ThreadPoolExecutor(max_workers=5) as executor:
                futures = {
                    executor.submit(self._classify_one, note): note
                    for note in batch
                }
                
                for future in as_completed(futures):
                    result = future.result()
                    results.append(result)
                    
        # 保存
        with open('data/classified_500.json', 'w') as f:
            json.dump(results, f, indent=2)
        
        return results
    
    def _classify_one(self, note):
        """1件分類"""
        with open(note['path'], 'r', encoding='utf-8') as f:
            content = f.read()[:1500]
        
        prompt = f"""
        Classify this note for Planetary OS:
        
        Output JSON:
        {{
            "domain": "MBT55/AGRIX/HealthBook/PBPE/Climate",
            "subdomain": "specific area",
            "has_equations": true/false,
            "has_data": true/false,
            "target_relevance": "Gates/Microsoft/Both/None",
            "summary": "one-line summary"
        }}
        
        Content: {content[:1000]}
        """
        
        response = self.client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[{"role": "user", "content": prompt}],
            response_format={"type": "json_object"}
        )
        
        return {
            "path": note['path'],
            "classification": json.loads(response.choices[0].message.content)
        }
```

```python
"""

from neo4j import GraphDatabase
import json

class CoreGraphBuilder:
    def __init__(self):
        self.driver = GraphDatabase.driver(
            "bolt://localhost:7687",
            auth=("neo4j", "password")
        )
    
    def build_graph(self):
        """一括グラフ構築"""
        
        with open('data/classified_500.json') as f:
            notes = json.load(f)
        
        with self.driver.session() as session:
            # 一括作成
            for note in notes[:100]:  # 100件ずつ
                path = note['path']
                cls = note['classification']
                
                session.run("""
                    CREATE (n:Note {
                        path: $path,
                        domain: $domain,
                        subdomain: $subdomain,
                        has_equations: $has_equations,
                        has_data: $has_data,
                        target: $target
                    })
                    """,
                    path=path,
                    domain=cls.get('domain', 'Unknown'),
                    subdomain=cls.get('subdomain', ''),
                    has_equations=cls.get('has_equations', False),
                    has_data=cls.get('has_data', False),
                    target=cls.get('target_relevance', 'None')
                )
            
            # インデックス作成
            session.run("CREATE INDEX domain_idx IF NOT EXISTS FOR (n:Note) ON (n.domain)")
            session.run("CREATE INDEX target_idx IF NOT EXISTS FOR (n:Note) ON (n.target)")
        
        print("✅ グラフ構築完了")
```

```python
"""
統合スーパーバイザー（全エージェント連携）
"""

from mbt55_agent import MBT55Agent
from agrix_agent import AGRIXAgent
from healthbook_agent import HealthBookAgent
from pbpe_agent import PBPEAgent

class IntegratedSupervisor:
    def __init__(self):
        self.agents = {
            "mbt55": MBT55Agent(),
            "agrix": AGRIXAgent(),
            "healthbook": HealthBookAgent(),
            "pbpe": PBPEAgent()
        }
        self.graph = self._load_graph()
    
    def _load_graph(self):
        """Neo4jからグラフ読み込み"""
        from neo4j import GraphDatabase
        driver = GraphDatabase.driver("bolt://localhost:7687", auth=("neo4j", "password"))
        return driver
    
    def query_with_context(self, query: str):
        """グラフを参照してエージェント実行"""
        
        # 1. 意図認識
        intent = self._detect_intent(query)
        
        # 2. 関連ノートをグラフ検索
        with self.graph.session() as session:
            result = session.run(
                "MATCH (n:Note) WHERE n.domain = $domain RETURN n.path LIMIT 5",
                domain=intent['domain']
            )
            contexts = [record['n.path'] for record in result]
        
        # 3. エージェント実行
        agent = self.agents[intent['agent']]
        result = agent.process(query)
        
        # 4. コンテキスト付きで返す
        return {
            "result": result,
            "context_sources": contexts,
            "intent": intent
        }
    
    def _detect_intent(self, query):
        """簡易意図認識"""
        q = query.lower()
        if 'mbt55' in q: return {"agent": "mbt55", "domain": "MBT55"}
        if 'agrix' in q: return {"agent": "agrix", "domain": "AGRIX"}
        if 'health' in q: return {"agent": "healthbook", "domain": "HealthBook"}
        if 'pbpe' in q or 'token' in q: return {"agent": "pbpe", "domain": "PBPE"}
        return {"agent": "mbt55", "domain": "MBT55"}  # デフォルト

# テスト実行
```
