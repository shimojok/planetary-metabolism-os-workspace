```python
"""
分類エージェント - 500核心ノートを4領域に分類
"""

import json
from pathlib import Path
import openai
from dotenv import load_dotenv
import os

load_dotenv()
client = openai.OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

# OpenAI APIキーが設定されているか確認
if not os.getenv("OPENAI_API_KEY"):
    print("❌ OPENAI_API_KEYが設定されていません。")
    print("📝 .envファイルを作成し、OPENAI_API_KEY=sk-xxx を記述してください。")
    exit(1)

class ClassifierAgent:
    def __init__(self):
        with open('data/core_500.json', 'r', encoding='utf-8') as f:
            self.core_data = json.load(f)
        self.notes = self.core_data['core_notes']
        self.classified = []
        
        # 分類先ドメイン
        self.domains = ["MBT55", "AGRIX", "HealthBook", "PBPE", "Other"]
    
    def classify_note(self, note):
        """1件のノートをAIで分類"""
        
        # ファイルの内容を読み込む（最初の1500文字）
        try:
            with open(note['path'], 'r', encoding='utf-8') as f:
                content = f.read(1500)
        except:
            content = ""
        
        prompt = f"""
        以下のノートを、Planetary Metabolic OSの知識体系において分類してください。
        
        分類先ドメイン：
        - MBT55: 微生物代謝、teleodynamics、微生物群集、発酵、炭素循環、GHG削減メカニズム
        - AGRIX: 農業政策、社会実装、土壌再生、農業経済、地域展開
        - HealthBook: 人体代謝、健康、腸内細菌、栄養、疾病リスク、予防医療
        - PBPE: 金融モデル、トークンエコノミクス、炭素クレジット、投資、ROI
        - Other: 上記以外
        
        出力はJSON形式で：
        {{
            "domain": "MBT55/AGRIX/HealthBook/PBPE/Other",
            "confidence": 0.0-1.0,
            "key_terms": ["主要キーワード1", "主要キーワード2"],
            "summary": "1行の要約"
        }}
        
        ノート情報：
        ファイル名: {note['name']}
        スコア: {note['score']}
        
        ノート内容（最初の1500文字）：
        {content}
        """
        
        try:
            response = client.chat.completions.create(
                model="gpt-4o-mini",
                messages=[{"role": "user", "content": prompt}],
                response_format={"type": "json_object"},
                temperature=0.3
            )
            result = json.loads(response.choices[0].message.content)
            result['path'] = note['path']
            result['score'] = note['score']
            result['name'] = note['name']
            return result
        except Exception as e:
            print(f"エラー: {e}")
            return {
                "domain": "Other",
                "confidence": 0.0,
                "key_terms": [],
                "summary": "分類エラー",
                "path": note['path'],
                "score": note['score'],
                "name": note['name']
            }
    
    def classify_all(self):
        """全ノートを分類"""
        total = len(self.notes)
        print(f"📊 全 {total} 件のノートを分類します")
        
        for i, note in enumerate(self.notes):
            print(f"📄 [{i+1}/{total}] {note['name'][:50]}...")
            result = self.classify_note(note)
            self.classified.append(result)
            
            # 10件ごとに中間保存
            if (i + 1) % 10 == 0:
                self.save_progress()
        
        self.save_final()
        return self.classified
    
    def save_progress(self):
        """途中経過を保存"""
        with open('data/classified_progress.json', 'w', encoding='utf-8') as f:
            json.dump(self.classified, f, indent=2, ensure_ascii=False)
        print(f"   💾 途中保存: {len(self.classified)}件")
    
    def save_final(self):
        """最終結果を保存"""
        with open('data/classified_500.json', 'w', encoding='utf-8') as f:
            json.dump(self.classified, f, indent=2, ensure_ascii=False)
        
        # 集計
        domain_counts = {}
        for item in self.classified:
            domain = item['domain']
            domain_counts[domain] = domain_counts.get(domain, 0) + 1
        
        print("\n" + "="*50)
        print("📊 分類結果サマリー")
        print("="*50)
        for domain, count in sorted(domain_counts.items(), key=lambda x: x[1], reverse=True):
            print(f"  {domain}: {count}件")
        
        # 分類結果を保存
        summary = {
            "total": len(self.classified),
            "domain_counts": domain_counts,
            "high_confidence": sum(1 for i in self.classified if i['confidence'] > 0.8)
        }
        with open('data/classification_summary.json', 'w', encoding='utf-8') as f:
            json.dump(summary, f, indent=2, ensure_ascii=False)
        
        print(f"\n✅ 高信頼度（80%以上）: {summary['high_confidence']}件")

if __name__ == "__main__":
    classifier = ClassifierAgent()
    classifier.classify_all()
```
